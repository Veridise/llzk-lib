28FailureOr<LLZKDialectVersion> LLZKDialectVersion::read(DialectBytecodeReader &reader) {
29 LLZKDialectVersion v;
30 if (failed(reader.readVarInt(v.majorVersion)) || failed(reader.readVarInt(v.minorVersion)) ||
