45int main(int argc, char **argv) {
46 llvm::sys::PrintStackTraceOnErrorSignal(llvm::StringRef());
47 llvm::setBugReportMsg("PLEASE submit a bug report to " BUG_REPORT_URL
48 " and include the crash backtrace, relevant LLZK files,"
49 " and associated run script(s).\n");
50 llvm::cl::AddExtraVersionPrinter([](llvm::raw_ostream &os) {
51 os << "\nLLZK (" LLZK_URL "):\n LLZK version " LLZK_VERSION_STRING "\n";
52 });
53
54 // MLIR initialization
55 mlir::DialectRegistry registry;
56 llzk::registerAllDialects(registry);
57 llzk::registerAnalysisPasses();
58 llzk::registerTransformationPasses();
59 llzk::array::registerTransformationPasses();
60 llzk::include::registerTransformationPasses();
61 llzk::polymorphic::registerTransformationPasses();
62 llzk::registerTransformationPassPipelines();
63 llzk::registerValidationPasses();
64
65 // Register and parse command line options.
66 std::string inputFilename, outputFilename;
67 std::tie(inputFilename, outputFilename) =
68 registerAndParseCLIOptions(argc, argv, "llzk-opt", registry);
69
70 if (PrintAllOps) {
71 mlir::MLIRContext context;
72 context.appendDialectRegistry(registry);
73 context.loadAllAvailableDialects();
74 llvm::outs() << "All ops registered in LLZK IR: {\n";
75 for (const auto &opName : context.getRegisteredOperations()) {
76 llvm::outs() << " " << opName.getStringRef() << '\n';
77 }
78 llvm::outs() << "}\n";
79 return EXIT_SUCCESS;
80 }
81
82 // Set the include directories from CL option
83 if (mlir::failed(llzk::GlobalSourceMgr::get().setup(IncludeDirs))) {
84 return EXIT_FAILURE;
85 }
86
87 // Run 'mlir-opt'
88 auto result = mlir::MlirOptMain(argc, argv, inputFilename, outputFilename, registry);
89 return asMainReturnCode(result);
90}
