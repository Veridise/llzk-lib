149inline void dumpSymbolTableWalk(mlir::Operation *symbolTableOp) {
150 std::string output; // buffer to avoid multi-threaded mess
151 llvm::raw_string_ostream oss(output);
152 oss << "Dumping symbol walk (self = [" << symbolTableOp << "]): \n";
153 auto walkFn = [&](mlir::Operation *op, bool allUsesVisible) {
154 oss << " found op [" << op << "] " << op->getName() << " named "
155 << op->getAttrOfType<mlir::StringAttr>(mlir::SymbolTable::getSymbolAttrName()) << '\n';
156 };
157 mlir::SymbolTable::walkSymbolTables(symbolTableOp, /*allSymUsesVisible=*/true, walkFn);
158 llvm::outs() << output;
159}

161inline void
162dumpSymbolTable(llvm::raw_ostream &stream, mlir::SymbolTable &symTab, unsigned indent = 0) {
163 for (unsigned i = 0; i < indent; ++i) {
164 stream << " ";
165 }
166 stream << "Dumping SymbolTable [" << &symTab << "]: \n";
167 auto *rawSymbolTablePtr = reinterpret_cast<char *>(&symTab);
168 auto *privateFieldPtr =
169 reinterpret_cast<llvm::DenseMap<mlir::Attribute, mlir::Operation *> *>(rawSymbolTablePtr + 8);
170 for (llvm::detail::DenseMapPair<mlir::Attribute, mlir::Operation *> &p : *privateFieldPtr) {
171 for (unsigned i = 0; i < indent; ++i) {
172 stream << ' ';
173 }
174 mlir::Operation *op = p.second;
175 stream << ' ' << p.first << " -> [" << op << "] " << op->getName() << '\n';
176 }
177}

186inline void dumpSymbolTables(llvm::raw_ostream &stream, mlir::SymbolTableCollection &tables) {
187 stream << "Dumping SymbolTableCollection [" << &tables << "]: \n";
188 auto *rawObjectPtr = reinterpret_cast<char *>(&tables);
189 auto *privateFieldPtr =
190 reinterpret_cast<llvm::DenseMap<mlir::Operation *, std::unique_ptr<mlir::SymbolTable>> *>(
191 rawObjectPtr + 0
192 );
193 for (llvm::detail::DenseMapPair<mlir::Operation *, std::unique_ptr<mlir::SymbolTable>> &p :
194 *privateFieldPtr) {
195 stream << " [" << p.first << "] " << p.first->getName() << " -> " << '\n';
196 dumpSymbolTable(stream, *p.second.get(), 2);
197 }
198}
