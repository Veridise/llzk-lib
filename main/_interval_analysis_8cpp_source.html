32const Field &Field::getField(const char *fieldName) {
33 static llvm::DenseMap<llvm::StringRef, Field> knownFields;
34 static std::once_flag fieldsInit;
35 std::call_once(fieldsInit, initKnownFields, knownFields);
36
37 if (auto it = knownFields.find(fieldName); it != knownFields.end()) {
38 return it->second;
39 }
40 llvm::report_fatal_error("field \"" + mlir::Twine(fieldName) + "\" is unsupported");

58llvm::APSInt Field::reduce(llvm::APSInt i) const {
59 unsigned maxBits = std::max(i.getBitWidth(), bitWidth());
60 llvm::APSInt m = (i.extend(maxBits) % prime().extend(maxBits)).trunc(bitWidth());
61 if (m.isNegative()) {
62 return prime() + m;
63 }
64 return m;
65}

68 auto ap = llvm::APSInt(llvm::APInt(bitWidth(), i));

74Interval UnreducedInterval::reduce(const Field &field) const {
75 if (safeGt(a, b)) {
76 return Interval::Empty(field);
77 }
78 if (safeGe(width(), field.prime())) {
79 return Interval::Entire(field);
80 }
81 auto lhs = field.reduce(a), rhs = field.reduce(b);
82 // lhs and rhs are now guaranteed to have the same bitwidth, so we can use
83 // built-in functions.
84 if ((rhs - lhs).isZero()) {
85 return Interval::Degenerate(field, lhs);
86 }
87
88 const auto &half = field.half();
89 if (lhs.ule(rhs)) {
90 if (lhs.ult(half) && rhs.ult(half)) {
91 return Interval::TypeA(field, lhs, rhs);
92 } else if (lhs.ult(half)) {
93 return Interval::TypeC(field, lhs, rhs);
94 } else {
95 return Interval::TypeB(field, lhs, rhs);
96 }
97 } else {
98 if (lhs.uge(half) && rhs.ult(half)) {
99 return Interval::TypeF(field, lhs, rhs);
100 } else {
101 return Interval::Entire(field);
102 }
103 }
104}

171bool UnreducedInterval::overlaps(const UnreducedInterval &rhs) const {
172 return isNotEmpty() && rhs.isNotEmpty() && safeGe(b, rhs.a) && safeLe(a, rhs.b);

175std::strong_ordering operator<=>(const UnreducedInterval &lhs, const UnreducedInterval &rhs) {
176 if (safeLt(lhs.a, rhs.a) || (safeEq(lhs.a, rhs.a) && safeLt(lhs.b, rhs.b))) {
177 return std::strong_ordering::less;
178 }
179 if (safeGt(lhs.a, rhs.a) || (safeEq(lhs.a, rhs.a) && safeGt(lhs.b, rhs.b))) {
180 return std::strong_ordering::greater;
181 }
182 return std::strong_ordering::equal;
183}

185llvm::APSInt UnreducedInterval::width() const {
186 llvm::APSInt w;
187 if (safeGt(a, b)) {
188 // This would be reduced to an empty Interval, so the width is just zero.
189 w = llvm::APSInt::getUnsigned(0);
190 } else {
192 w = expandingSub(b, a)++;
193 }
194 ensure(safeGe(w, llvm::APSInt::getUnsigned(0)), "cannot have negative width");
195 return w;
196}

224Interval Interval::join(const Interval &rhs) const {
225 auto &lhs = *this;
226 ensure(
227 lhs.getField() == rhs.getField(), "interval operations across differing fields is unsupported"
228 );
229 const Field &f = lhs.getField();
230
231 // Trivial cases
232 if (lhs.isEntire() || rhs.isEntire()) {
233 return Interval::Entire(f);
234 }
235 if (lhs.isEmpty()) {
236 return rhs;
237 }
238 if (rhs.isEmpty()) {
239 return lhs;
240 }
241 if (lhs.isDegenerate() || rhs.isDegenerate()) {
242 return lhs.toUnreduced().doUnion(rhs.toUnreduced()).reduce(f);
243 }
244
245 // More complex cases
246 if (areOneOf<
247 {Type::TypeA, Type::TypeA}, {Type::TypeB, Type::TypeB}, {Type::TypeC, Type::TypeC},
248 {Type::TypeA, Type::TypeC}, {Type::TypeB, Type::TypeC}>(lhs, rhs)) {
249 return Interval(rhs.ty, f, std::min(lhs.a, rhs.a), std::max(lhs.b, rhs.b));
250 }
251 if (areOneOf<{Type::TypeA, Type::TypeB}>(lhs, rhs)) {
252 auto lhsUnred = lhs.firstUnreduced();
253 auto opt1 = rhs.firstUnreduced().doUnion(lhsUnred);
254 auto opt2 = rhs.secondUnreduced().doUnion(lhsUnred);
255 if (opt1.width() <= opt2.width()) {
256 return opt1.reduce(f);
257 }
258 return opt2.reduce(f);
259 }
260 if (areOneOf<{Type::TypeF, Type::TypeF}, {Type::TypeA, Type::TypeF}>(lhs, rhs)) {
261 return lhs.firstUnreduced().doUnion(rhs.firstUnreduced()).reduce(f);
262 }
263 if (areOneOf<{Type::TypeB, Type::TypeF}>(lhs, rhs)) {
264 return lhs.secondUnreduced().doUnion(rhs.firstUnreduced()).reduce(f);
265 }
266 if (areOneOf<{Type::TypeC, Type::TypeF}>(lhs, rhs)) {
267 return Interval::Entire(f);
268 }
269 if (areOneOf<
270 {Type::TypeB, Type::TypeA}, {Type::TypeC, Type::TypeA}, {Type::TypeC, Type::TypeB},
271 {Type::TypeF, Type::TypeA}, {Type::TypeF, Type::TypeB}, {Type::TypeF, Type::TypeC}>(
272 lhs, rhs
273 )) {
274 return rhs.join(lhs);
275 }
276 llvm::report_fatal_error("unhandled join case");
277 return Interval::Entire(f);
278}

280Interval Interval::intersect(const Interval &rhs) const {
281 auto &lhs = *this;
282 ensure(
283 lhs.getField() == rhs.getField(), "interval operations across differing fields is unsupported"
284 );
285
286 // Trivial cases
287 if (lhs.isEmpty() || rhs.isEmpty()) {
288 return Interval::Empty(field.get());
289 }
290 if (lhs.isEntire()) {
291 return rhs;
292 }
293 if (rhs.isEntire()) {
294 return lhs;
295 }
296 if (lhs.isDegenerate() || rhs.isDegenerate()) {
297 return lhs.toUnreduced().intersect(rhs.toUnreduced()).reduce(field.get());
298 }
299
300 // More complex cases
301 if (areOneOf<
302 {Type::TypeA, Type::TypeA}, {Type::TypeB, Type::TypeB}, {Type::TypeC, Type::TypeC},
303 {Type::TypeA, Type::TypeC}, {Type::TypeB, Type::TypeC}>(lhs, rhs)) {
304 auto maxA = std::max(lhs.a, rhs.a);
305 auto minB = std::min(lhs.b, rhs.b);
306 if (maxA <= minB) {
307 return Interval(lhs.ty, field.get(), maxA, minB);
308 } else {
309 return Interval::Empty(field.get());
310 }
311 }
312 if (areOneOf<{Type::TypeA, Type::TypeB}>(lhs, rhs)) {
313 return Interval::Empty(field.get());
314 }
315 if (areOneOf<{Type::TypeF, Type::TypeF}, {Type::TypeA, Type::TypeF}>(lhs, rhs)) {
316 return lhs.firstUnreduced().intersect(rhs.firstUnreduced()).reduce(field.get());
317 }
318 if (areOneOf<{Type::TypeB, Type::TypeF}>(lhs, rhs)) {
319 return lhs.secondUnreduced().intersect(rhs.firstUnreduced()).reduce(field.get());
320 }
321 if (areOneOf<{Type::TypeC, Type::TypeF}>(lhs, rhs)) {
322 auto rhsUnred = rhs.firstUnreduced();
323 auto opt1 = lhs.firstUnreduced().intersect(rhsUnred).reduce(field.get());
324 auto opt2 = lhs.secondUnreduced().intersect(rhsUnred).reduce(field.get());
325 ensure(!opt1.isEntire() && !opt2.isEntire(), "impossible intersection");
326 if (opt1.isEmpty()) {
327 return opt2;
328 }
329 if (opt2.isEmpty()) {
330 return opt1;
331 }
332 return opt1.join(opt2);
333 }
334 if (areOneOf<
335 {Type::TypeB, Type::TypeA}, {Type::TypeC, Type::TypeA}, {Type::TypeC, Type::TypeB},
336 {Type::TypeF, Type::TypeA}, {Type::TypeF, Type::TypeB}, {Type::TypeF, Type::TypeC}>(
337 lhs, rhs
338 )) {
339 return rhs.intersect(lhs);
340 }
341 return Interval::Empty(field.get());
342}

344Interval Interval::difference(const Interval &other) const {
345 // intersect checks that we're in the same field
346 Interval intersection = intersect(other);
347 if (intersection.isEmpty()) {
348 // There's nothing to remove, so just return this
349 return *this;
350 }
351
352 const Field &f = field.get();
353
354 // Trivial cases with a non-empty intersection
355 if (isDegenerate() || other.isEntire()) {
356 return Interval::Empty(f);
357 }
358 if (isEntire()) {
359 // Since we don't support punching arbitrary holes in ranges, we only reduce
360 // entire ranges if other is [0, b] or [a, prime - 1]
361 if (other.a == f.zero()) {
362 return UnreducedInterval(other.b + f.one(), f.maxVal()).reduce(f);
363 }
364 if (other.b == f.maxVal()) {
365 return UnreducedInterval(f.zero(), other.a - f.one()).reduce(f);
366 }
367
368 return *this;
369 }
370
371 // Non-trivial cases
372 // - Internal+internal or external+external cases
373 if ((is<Type::TypeA, Type::TypeB, Type::TypeC>() &&
374 intersection.is<Type::TypeA, Type::TypeB, Type::TypeC>()) ||
375 areOneOf<{Type::TypeF, Type::TypeF}>(*this, intersection)) {
376 // The intersection needs to be at the end of the interval, otherwise we would
377 // split the interval in two, and we aren't set up to support multiple intervals
378 // per value.
379 if (a != intersection.a && b != intersection.b) {
380 return *this;
381 }
382 // Otherwise, remove the intersection and reduce
383 if (a == intersection.a) {
384 return UnreducedInterval(intersection.b + f.one(), b).reduce(f);
385 }
386 // else b == intersection.b
387 return UnreducedInterval(a, intersection.a - f.one()).reduce(f);
388 }
389 // - Mixed internal/external cases. We flip the comparison
390 if (isTypeF()) {
391 if (a != intersection.b && b != intersection.a) {
392 return *this;
393 }
394 // Otherwise, remove the intersection and reduce
395 if (a == intersection.b) {
396 return UnreducedInterval(intersection.a + f.one(), b).reduce(f);
397 }
398 // else b == intersection.a
399 return UnreducedInterval(a, intersection.b - f.one()).reduce(f);
400 }
401
402 // In cases we don't know how to handle, we over-approximate and return
403 // the original interval.
404 return *this;
405}

416Interval operator*(const Interval &lhs, const Interval &rhs) {
417 ensure(lhs.field.get() == rhs.field.get(), "cannot multiply intervals in different fields");
418 const auto &field = lhs.field.get();
419 auto zeroInterval = Interval::Degenerate(field, field.zero());
420 if (lhs == zeroInterval || rhs == zeroInterval) {
421 return zeroInterval;
422 }
423 if (lhs.isEmpty() || rhs.isEmpty()) {
424 return Interval::Empty(field);
425 }
426 if (lhs.isEntire() || rhs.isEntire()) {
427 return Interval::Entire(field);
428 }
429
430 if (Interval::areOneOf<{Interval::Type::TypeB, Interval::Type::TypeB}>(lhs, rhs)) {
431 return (lhs.secondUnreduced() * rhs.secondUnreduced()).reduce(field);
432 }
433 return (lhs.firstUnreduced() * rhs.firstUnreduced()).reduce(field);
434}

436FailureOr<Interval> operator/(const Interval &lhs, const Interval &rhs) {
437 ensure(lhs.getField() == rhs.getField(), "cannot divide intervals in different fields");
438 const auto &field = rhs.getField();
439 if (rhs.width() > field.one()) {
440 return Interval::Entire(field);
441 }
442 if (rhs.a.isZero()) {
443 return failure();
444 }
445 return success(UnreducedInterval(lhs.a / rhs.a, lhs.b / rhs.a).reduce(field));
446}

534ExpressionValue
535cmp(llvm::SMTSolverRef solver, CmpOp op, const ExpressionValue &lhs, const ExpressionValue &rhs) {
536 ExpressionValue res;
537 res.i = Interval::Boolean(lhs.getField());
538 switch (op.getPredicate()) {
539 case FeltCmpPredicate::EQ:
540 res.expr = solver->mkEqual(lhs.expr, rhs.expr);
541 res.i = lhs.i.intersect(rhs.i);
542 break;
543 case FeltCmpPredicate::NE:
544 res.expr = solver->mkNot(solver->mkEqual(lhs.expr, rhs.expr));
545 break;
546 case FeltCmpPredicate::LT:
547 res.expr = solver->mkBVUlt(lhs.expr, rhs.expr);
548 break;
549 case FeltCmpPredicate::LE:
550 res.expr = solver->mkBVUle(lhs.expr, rhs.expr);
551 break;
552 case FeltCmpPredicate::GT:
553 res.expr = solver->mkBVUgt(lhs.expr, rhs.expr);
554 break;
555 case FeltCmpPredicate::GE:
556 res.expr = solver->mkBVUge(lhs.expr, rhs.expr);
557 break;
558 }
559 return res;
560}

562ExpressionValue fallbackBinaryOp(
563 llvm::SMTSolverRef solver, Operation *op, const ExpressionValue &lhs, const ExpressionValue &rhs
564) {
565 ExpressionValue res;
566 res.i = Interval::Entire(lhs.getField());
567 res.expr = TypeSwitch<Operation *, llvm::SMTExprRef>(op)
568 .Case<AndFeltOp>([&](AndFeltOp _) { return solver->mkBVAnd(lhs.expr, rhs.expr); })
569 .Case<OrFeltOp>([&](OrFeltOp _) { return solver->mkBVOr(lhs.expr, rhs.expr); })
570 .Case<XorFeltOp>([&](XorFeltOp _) { return solver->mkBVXor(lhs.expr, rhs.expr); })
571 .Case<ShlFeltOp>([&](ShlFeltOp _) { return solver->mkBVShl(lhs.expr, rhs.expr); })
572 .Case<ShrFeltOp>([&](ShrFeltOp _) {
573 return solver->mkBVLshr(lhs.expr, rhs.expr);
574 }).Default([&](Operation *unsupported) {
575 llvm::report_fatal_error(
576 "no fallback provided for " + mlir::Twine(unsupported->getName().getStringRef())
577 );
578 return nullptr;
579 });
580
581 return res;
582}

606ExpressionValue
607fallbackUnaryOp(llvm::SMTSolverRef solver, Operation *op, const ExpressionValue &val) {
608 const Field &field = val.getField();
609 ExpressionValue res;
610 res.i = Interval::Entire(field);
611 res.expr = TypeSwitch<Operation *, llvm::SMTExprRef>(op)
612 .Case<InvFeltOp>([&](InvFeltOp _) {
613 // The definition of an inverse X^-1 is Y s.t. XY % prime = 1.
614 // To create this expression, we create a new symbol for Y and add the
615 // XY % prime = 1 constraint to the solver.
616 std::string symName;
617 llvm::raw_string_ostream ss(symName);
618 ss << *op;
619
620 llvm::SMTExprRef invSym = field.createSymbol(solver, symName.c_str());
621 llvm::SMTExprRef one = solver->mkBitvector(field.one(), field.bitWidth());
622 llvm::SMTExprRef prime = solver->mkBitvector(field.prime(), field.bitWidth());
623 llvm::SMTExprRef mult = solver->mkBVMul(val.getExpr(), invSym);
624 llvm::SMTExprRef mod = solver->mkBVURem(mult, prime);
625 llvm::SMTExprRef constraint = solver->mkEqual(mod, one);
626 solver->addConstraint(constraint);
627 return invSym;
628 }).Default([&](Operation *unsupported) {
629 llvm::report_fatal_error(
630 "no fallback provided for " + mlir::Twine(op->getName().getStringRef())
631 );

734void IntervalDataFlowAnalysis::visitCallControlFlowTransfer(
735 CallOpInterface call, dataflow::CallControlFlowAction action,
736 const IntervalAnalysisLattice &before, IntervalAnalysisLattice *after
737) {
741 if (action == dataflow::CallControlFlowAction::EnterCallee) {
742 // We skip updating the incoming lattice for function calls,
743 // as values are relative to the containing function/struct, so we don't need to pollute
744 // the callee with the callers values.
745 setToEntryState(after);
746 }
750 else if (action == dataflow::CallControlFlowAction::ExitCallee) {
751 // Get the argument values of the lattice by getting the state as it would
752 // have been for the callsite.
753 dataflow::AbstractDenseLattice *beforeCall = nullptr;
754 if (auto *prev = call->getPrevNode()) {
755 beforeCall = getLattice(prev);
756 } else {
757 beforeCall = getLattice(call->getBlock());
758 }
759 ensure(beforeCall, "could not get prior lattice");
760
761 // The lattice at the return is the lattice before the call
762 propagateIfChanged(after, after->join(*beforeCall));
763 }
768 else if (action == mlir::dataflow::CallControlFlowAction::ExternalCallee) {
769 // For external calls, we propagate what information we already have from
770 // before the call to after the call, since the external call won't invalidate
771 // any of that information. It also, conservatively, makes no assumptions about
772 // external calls and their computation, so CDG edges will not be computed over
773 // input arguments to external functions.
774 join(after, before);
775 }

778void IntervalDataFlowAnalysis::visitOperation(
779 Operation *op, const Lattice &before, Lattice *after
780) {
781 ChangeResult changed = after->join(before);
782
783 llvm::SmallVector<LatticeValue> operandVals;
784
785 auto constrainRefLattice = dataflowSolver.lookupState<ConstrainRefLattice>(op);
786 ensure(constrainRefLattice, "failed to get lattice");
787
788 for (auto &operand : op->getOpOperands()) {
789 auto val = operand.get();
790 // First, lookup the operand value in the before state.
791 auto priorState = before.getValue(val);
792 if (succeeded(priorState)) {
793 operandVals.push_back(*priorState);
794 continue;
795 }
796
797 // Else, look up the stored value by constrain ref.
798 // We only care about scalar type values, so we ignore composite types, which
799 // are currently limited to non-Signal structs and arrays.
800 Type valTy = val.getType();
801 if (mlir::isa<ArrayType, StructType>(valTy) && !isSignalType(valTy)) {
802 operandVals.push_back(LatticeValue());
803 continue;
804 }
805
806 ConstrainRefLatticeValue refSet = constrainRefLattice->getOrDefault(val);
807 ensure(refSet.isScalar(), "should have ruled out array values already");
808
809 if (refSet.getScalarValue().empty()) {
810 // If we can't compute the reference, then there must be some unsupported
811 // op the reference analysis cannot handle. We emit a warning and return
812 // early, since there's no meaningful computation we can do for this op.
813 std::string warning;
814 debug::Appender(warning
815 ) << "state of "
816 << val << " is empty; defining operation is unsupported by constrain ref analysis";
817 op->emitWarning(warning);
818 propagateIfChanged(after, changed);
819 return;
820 } else if (!refSet.isSingleValue()) {
821 std::string warning;
822 debug::Appender(warning) << "operand " << val << " is not a single value " << refSet
823 << ", overapproximating";
824 op->emitWarning(warning);
825 operandVals.push_back(LatticeValue());
826 } else {
827 auto ref = refSet.getSingleValue();
828 ExpressionValue exprVal(field.get(), getOrCreateSymbol(ref));
829 changed |= after->setValue(val, exprVal);
830 operandVals.emplace_back(exprVal);
831 }
832 }
833
834 // Now, the way we update is dependent on the type of the operation.
835 if (!isConsideredOp(op)) {
836 op->emitWarning("unconsidered operation type, analysis may be incomplete");
837 }
838
839 if (isConstOp(op)) {
840 auto constVal = getConst(op);
841 auto expr = createConstBitvectorExpr(constVal);
842 ExpressionValue latticeVal(field.get(), expr, constVal);
843 changed |= after->setValue(op->getResult(0), latticeVal);
844 } else if (isArithmeticOp(op)) {
845 ensure(operandVals.size() <= 2, "arithmetic op with the wrong number of operands");
846 ExpressionValue result;
847 if (operandVals.size() == 2) {
848 result = performBinaryArithmetic(op, operandVals[0], operandVals[1]);
849 } else {
850 result = performUnaryArithmetic(op, operandVals[0]);
851 }
852
853 changed |= after->setValue(op->getResult(0), result);
854 } else if (EmitEqualityOp emitEq = mlir::dyn_cast<EmitEqualityOp>(op)) {
855 ensure(operandVals.size() == 2, "constraint op with the wrong number of operands");
856 Value lhsVal = emitEq.getLhs(), rhsVal = emitEq.getRhs();
857 ExpressionValue lhsExpr = operandVals[0].getScalarValue();
858 ExpressionValue rhsExpr = operandVals[1].getScalarValue();
859
860 ExpressionValue constraint = intersection(smtSolver, lhsExpr, rhsExpr);
861 // Update the LHS and RHS to the same value, but restricted intervals
862 // based on the constraints
863 changed |= applyInterval(emitEq, after, lhsVal, constraint.getInterval());
864 changed |= applyInterval(emitEq, after, rhsVal, constraint.getInterval());
865 changed |= after->addSolverConstraint(constraint);
866 } else if (AssertOp assertOp = mlir::dyn_cast<AssertOp>(op)) {
867 ensure(operandVals.size() == 1, "assert op with the wrong number of operands");
868 // assert enforces that the operand is true. So we apply an interval of [1, 1]
869 // to the operand.
870 changed |= applyInterval(
871 assertOp, after, assertOp.getCondition(),
872 Interval::Degenerate(field.get(), field.get().one())
873 );
874 // Also add the solver constraint that the expression must be true.
875 auto assertExpr = operandVals[0].getScalarValue();
876 changed |= after->addSolverConstraint(assertExpr);
877 } else if (auto readf = mlir::dyn_cast<FieldReadOp>(op);
878 readf && isSignalType(readf.getComponent().getType())) {
879 // The reg value read from the signal type is equal to the value of the Signal
880 // struct overall.
881 changed |= after->setValue(readf.getVal(), operandVals[0].getScalarValue());
882 } else if (!isReadOp(op) /* We do not need to explicitly handle read ops
883 since they are resolved at the operand value step where constrain refs are
884 queries (with the exception of the Signal struct, see above). */
885 && !isReturnOp(op) /* We do not currently handle return ops as the analysis
886 is currently limited to constrain functions, which return no value. */
887 && !isDefinitionOp(op) /* The analysis ignores definition ops. */
888 &&
889 !mlir::isa<CreateStructOp>(op) /* We do not need to analyze the creation of structs. */
890 ) {
891 op->emitWarning("unhandled operation, analysis may be incomplete");
892 }
893
894 propagateIfChanged(after, changed);
895}

999ChangeResult IntervalDataFlowAnalysis::applyInterval(
1000 Operation *originalOp, Lattice *after, Value val, Interval newInterval
1001) {
1002 auto latValRes = after->getValue(val);
1003 if (failed(latValRes)) {
1004 // visitOperation didn't add val to the lattice, so there's nothing to do
1005 return ChangeResult::NoChange;
1006 }
1007 ExpressionValue newLatticeVal = latValRes->getScalarValue().withInterval(newInterval);
1008 ChangeResult res = after->setValue(val, newLatticeVal);
1009 // To allow the dataflow analysis to do its fixed-point iteration, we need to
1010 // add the new expression to val's lattice as well.
1011 Lattice *valLattice = nullptr;
1012 if (auto valOp = val.getDefiningOp()) {
1013 // Getting the lattice at valOp gives us the "after" lattice, but we want to
1014 // update the "before" lattice so that the inputs to visitOperation will be
1015 // changed.
1016 if (auto prev = valOp->getPrevNode()) {
1017 valLattice = getOrCreate<Lattice>(prev);
1018 } else {
1019 valLattice = getOrCreate<Lattice>(valOp->getBlock());
1020 }
1021 } else if (auto blockArg = mlir::dyn_cast<BlockArgument>(val)) {
1022 valLattice = getOrCreate<Lattice>(blockArg.getOwner());
1023 } else {
1024 valLattice = getOrCreate<Lattice>(val);
1025 }
1026 ensure(valLattice, "val should have a lattice");
1027 if (valLattice != after) {
1028 propagateIfChanged(valLattice, valLattice->setValue(val, newLatticeVal));
1029 }
1030
1031 // Now we descend into val's operands, if it has any.
1032 Operation *definingOp = val.getDefiningOp();
1033 if (!definingOp) {
1034 return res;
1035 }
1036
1037 const Field &f = field.get();
1038
1039 // This is a rules-based operation. If we have a rule for a given operation,
1040 // then we can make some kind of update, otherwise we leave the intervals
1041 // as is.
1042 // - First we'll define all the rules so the type switch can be less messy
1043
1044 // cmp.<pred> restricts each side of the comparison if the result is known.
1045 auto cmpCase = [&](CmpOp cmpOp) {
1046 // Cmp output range is [0, 1], so in order to do something, we must have newInterval
1047 // either "true" (1) or "false" (0)
1048 Interval maxInterval = Interval::Boolean(f);
1049 ensure(
1050 newInterval.intersect(maxInterval).isNotEmpty(),
1051 "new interval for CmpOp outside of allowed boolean range or is empty"
1052 );
1053 if (!newInterval.isDegenerate()) {
1054 // The comparison result is unknown, so we can't update the operand ranges
1055 return ChangeResult::NoChange;
1056 }
1057
1058 bool cmpTrue = newInterval.rhs() == f.one();
1059
1060 Value lhs = cmpOp->getOperand(0), rhs = cmpOp->getOperand(1);
1061 auto lhsLatValRes = after->getValue(lhs), rhsLatValRes = after->getValue(rhs);
1062 if (failed(lhsLatValRes) || failed(rhsLatValRes)) {
1063 return ChangeResult::NoChange;
1064 }
1065 ExpressionValue lhsExpr = lhsLatValRes->getScalarValue(),
1066 rhsExpr = rhsLatValRes->getScalarValue();
1067
1068 Interval newLhsInterval, newRhsInterval;
1069 const Interval &lhsInterval = lhsExpr.getInterval();
1070 const Interval &rhsInterval = rhsExpr.getInterval();
1071
1072 FeltCmpPredicate pred = cmpOp.getPredicate();
1073 // predicate cases
1074 auto eqCase = [&]() {
1075 return (pred == FeltCmpPredicate::EQ && cmpTrue) ||
1076 (pred == FeltCmpPredicate::NE && !cmpTrue);
1077 };
1078 auto neCase = [&]() {
1079 return (pred == FeltCmpPredicate::NE && cmpTrue) ||
1080 (pred == FeltCmpPredicate::EQ && !cmpTrue);
1081 };
1082 auto ltCase = [&]() {
1083 return (pred == FeltCmpPredicate::LT && cmpTrue) ||
1084 (pred == FeltCmpPredicate::GE && !cmpTrue);
1085 };
1086 auto leCase = [&]() {
1087 return (pred == FeltCmpPredicate::LE && cmpTrue) ||
1088 (pred == FeltCmpPredicate::GT && !cmpTrue);
1089 };
1090 auto gtCase = [&]() {
1091 return (pred == FeltCmpPredicate::GT && cmpTrue) ||
1092 (pred == FeltCmpPredicate::LE && !cmpTrue);
1093 };
1094 auto geCase = [&]() {
1095 return (pred == FeltCmpPredicate::GE && cmpTrue) ||
1096 (pred == FeltCmpPredicate::LT && !cmpTrue);
1097 };
1098
1099 // new intervals based on case
1100 if (eqCase()) {
1101 newLhsInterval = newRhsInterval = lhsInterval.intersect(rhsInterval);
1102 } else if (neCase()) {
1103 if (lhsInterval.isDegenerate() && rhsInterval.isDegenerate() && lhsInterval == rhsInterval) {
1104 // In this case, we know lhs and rhs cannot satisfy this assertion, so they have
1105 // an empty value range.
1106 newLhsInterval = newRhsInterval = Interval::Empty(f);
1107 } else {
1108 // Leave unchanged
1109 newLhsInterval = lhsInterval;
1110 newRhsInterval = rhsInterval;
1111 }
1112 } else if (ltCase()) {
1113 newLhsInterval = lhsInterval.toUnreduced().computeLTPart(rhsInterval.toUnreduced()).reduce(f);
1114 newRhsInterval = rhsInterval.toUnreduced().computeGEPart(lhsInterval.toUnreduced()).reduce(f);
1115 } else if (leCase()) {
1116 newLhsInterval = lhsInterval.toUnreduced().computeLEPart(rhsInterval.toUnreduced()).reduce(f);
1117 newRhsInterval = rhsInterval.toUnreduced().computeGTPart(lhsInterval.toUnreduced()).reduce(f);
1118 } else if (gtCase()) {
1119 newLhsInterval = lhsInterval.toUnreduced().computeGTPart(rhsInterval.toUnreduced()).reduce(f);
1120 newRhsInterval = rhsInterval.toUnreduced().computeLEPart(lhsInterval.toUnreduced()).reduce(f);
1121 } else if (geCase()) {
1122 newLhsInterval = lhsInterval.toUnreduced().computeGEPart(rhsInterval.toUnreduced()).reduce(f);
1123 newRhsInterval = rhsInterval.toUnreduced().computeLTPart(lhsInterval.toUnreduced()).reduce(f);
1124 } else {
1125 cmpOp->emitWarning("unhandled cmp predicate");
1126 return ChangeResult::NoChange;
1127 }
1128
1129 // Now we recurse to each operand
1130 return applyInterval(originalOp, after, lhs, newLhsInterval) |
1131 applyInterval(originalOp, after, rhs, newRhsInterval);
1132 };
1133
1134 // If the result of a multiplication is non-zero, then both operands must be
1135 // non-zero.
1136 auto mulCase = [&](MulFeltOp mulOp) {
1137 auto zeroInt = Interval::Degenerate(f, f.zero());
1138 if (newInterval.intersect(zeroInt).isNotEmpty()) {
1139 // The multiplication may be zero, so we can't reduce the operands to be non-zero
1140 return ChangeResult::NoChange;
1141 }
1142
1143 Value lhs = mulOp->getOperand(0), rhs = mulOp->getOperand(1);
1144 auto lhsLatValRes = after->getValue(lhs), rhsLatValRes = after->getValue(rhs);
1145 if (failed(lhsLatValRes) || failed(rhsLatValRes)) {
1146 return ChangeResult::NoChange;
1147 }
1148 ExpressionValue lhsExpr = lhsLatValRes->getScalarValue(),
1149 rhsExpr = rhsLatValRes->getScalarValue();
1150 Interval newLhsInterval = lhsExpr.getInterval().difference(zeroInt);
1151 Interval newRhsInterval = rhsExpr.getInterval().difference(zeroInt);
1152 return applyInterval(originalOp, after, lhs, newLhsInterval) |
1153 applyInterval(originalOp, after, rhs, newRhsInterval);
1154 };
1155
1156 // We have a special case for the Signal struct: if this value is created
1157 // from reading a Signal struct's reg field, we also apply the interval to
1158 // the struct itself.
1159 auto readfCase = [&](FieldReadOp readfOp) {
1160 Value comp = readfOp.getComponent();
1161 if (isSignalType(comp.getType())) {
1162 return applyInterval(originalOp, after, comp, newInterval);
1163 }
1164 return ChangeResult::NoChange;
1165 };
1166
1167 // - Apply the rules given the op.
1168 // NOTE: disabling clang-format for this because it makes the last case statement
1169 // look ugly.
1170 // clang-format off
1171 res |= TypeSwitch<Operation *, ChangeResult>(definingOp)
1172 .Case<CmpOp>([&](CmpOp op) { return cmpCase(op); })
1173 .Case<MulFeltOp>([&](MulFeltOp op) { return mulCase(op); })
1174 .Case<FieldReadOp>([&](FieldReadOp op){ return readfCase(op); })
1175 .Default([&](Operation *_) { return ChangeResult::NoChange; });
1176 // clang-format on
1177
1178 return res;
1179}

1183LogicalResult StructIntervals::computeIntervals(
1184 mlir::DataFlowSolver &solver, mlir::AnalysisManager &am, IntervalAnalysisContext &ctx
1185) {
1186 // Get the lattice at the end of the constrain function.
1187 function::ReturnOp constrainEnd;
1188 structDef.getConstrainFuncOp().walk([&constrainEnd](function::ReturnOp r) mutable {
1189 constrainEnd = r;
1190 });
1191
1192 const IntervalAnalysisLattice *constrainLattice =
1193 solver.lookupState<IntervalAnalysisLattice>(constrainEnd);
1194
1195 constrainSolverConstraints = constrainLattice->getConstraints();
1196
1197 for (const auto &ref : ConstrainRef::getAllConstrainRefs(structDef)) {
1198 // We only want to compute intervals for field elements and not composite types,
1199 // with the exception of the Signal struct.
1200 if (!ref.isScalar() && !ref.isSignal()) {
1201 continue;
1202 }
1203 // We also don't want to show the interval for a Signal and its internal reg.
1204 if (auto parentOr = ref.getParentPrefix(); succeeded(parentOr) && parentOr->isSignal()) {
1205 continue;
1206 }
1207 auto symbol = ctx.getSymbol(ref);
1208 auto constrainInterval = constrainLattice->findInterval(symbol);
1209 if (succeeded(constrainInterval)) {
1210 constrainFieldRanges[ref] = *constrainInterval;
1211 } else {
1212 constrainFieldRanges[ref] = Interval::Entire(ctx.field);
1213 }
1214 }
1215
1216 return success();
1217}
