17/// 2. Run a dialect conversion that does the following:
18///
19/// - Replace `FieldReadOp` and `FieldWriteOp` targeting the fields that were split in step 1 so
20/// they instead perform scalar reads and writes from the new fields. The transformation is
21/// local to the current op. Therefore, when replacing the `FieldReadOp` a new array is created
22/// locally and all uses of the `FieldReadOp` are replaced with the new array Value, then each
23/// scalar field read is followed by scalar write into the new array. Similarly, when replacing
24/// a `FieldWriteOp`, each element in the array operand needs a scalar read from the array
25/// followed by a scalar write to the new field. Making only local changes keeps this step
26/// simple and later steps will optimize.
27///
28/// - Replace `ArrayLengthOp` with the constant size of the selected dimension.
29///
30/// - Remove element initialization from `CreateArrayOp` and instead insert a list of
31/// `WriteArrayOp` immediately following.
32///
33/// - Desugar `InsertArrayOp` and `ExtractArrayOp` into their element-wise scalar reads/writes.
34///
35/// - Split arrays to scalars in `FuncDefOp`, `CallOp`, and `ReturnOp` and insert the necessary
36/// create/read/write ops so the changes are as local as possible (just as described for
37/// `FieldReadOp` and `FieldWriteOp`)
38///
39/// 3. Run MLIR "sroa" pass to split each array with linear size `N` into `N` arrays of size 1 (to
40/// prepare for "mem2reg" pass because it's API does not allow for indexing to split aggregates).
41///
42/// 4. Run MLIR "mem2reg" pass to convert all of the size 1 array allocation and access into SSA
43/// values. This pass also runs several standard optimizations so the final result is condensed.
44///
45/// Note: This transformation imposes a "last write wins" semantics on array elements. If
46/// different/configurable semantics are added in the future, some additional transformation would
47/// be necessary before/during this pass so that multiple writes to the same index can be handled
48/// properly while they still exist.
49///
50/// Note: This transformation will introduce an undef op when there exists a read from an array
51/// index that was not earlier written to.
52///

87inline ArrayType splittableArray(ArrayType at) { return at.hasStaticShape() ? at : nullptr; }

90inline ArrayType splittableArray(Type t) {

99inline bool containsSplittableArrayType(Type t) {
100 return t
101 .walk([](ArrayType a) {
102 return splittableArray(a) ? WalkResult::interrupt() : WalkResult::skip();

107template <typename T> bool containsSplittableArrayType(ValueTypeRange<T> types) {

118void splitArrayTypeTo(Type t, SmallVector<Type> &collect) {
119 if (ArrayType at = splittableArray(t)) {
120 int64_t n = at.getNumElements();
121 assert(std::cmp_less_equal(n, std::numeric_limits<SmallVector<Type>::size_type>::max()));
122 collect.append(n, at.getElementType());
123 } else {
124 collect.push_back(t);
125 }
126}

129template <typename TypeCollection>
130inline void splitArrayTypeTo(TypeCollection types, SmallVector<Type> &collect) {
131 for (Type t : types) {
132 splitArrayTypeTo(t, collect);
133 }

138template <typename TypeCollection> inline SmallVector<Type> splitArrayType(TypeCollection types) {
139 SmallVector<Type> collect;
140 splitArrayTypeTo(types, collect);

169CallOp newCallOpWithSplitResults(
170 CallOp oldCall, CallOp::Adaptor adaptor, ConversionPatternRewriter &rewriter
171) {
172 OpBuilder::InsertionGuard guard(rewriter);
173 rewriter.setInsertionPointAfter(oldCall);
174
175 Operation::result_range oldResults = oldCall.getResults();
176 CallOp newCall = rewriter.create<CallOp>(
177 oldCall.getLoc(), splitArrayType(oldResults.getTypes()), oldCall.getCallee(),
178 adaptor.getArgOperands()
179 );
180
181 auto newResults = newCall.getResults().begin();
182 for (Value oldVal : oldResults) {
183 if (ArrayType at = splittableArray(oldVal.getType())) {
184 Location loc = oldVal.getLoc();
185 // Generate `CreateArrayOp` and replace uses of the result with it.
186 auto newArray = rewriter.create<CreateArrayOp>(loc, at);
187 rewriter.replaceAllUsesWith(oldVal, newArray);
188
189 // For all indices in the ArrayType (i.e. the element count), write the next
190 // result from the new CallOp to the new array.
191 std::optional<SmallVector<ArrayAttr>> allIndices = at.getSubelementIndices();
192 assert(allIndices); // follows from legal() check
193 assert(std::cmp_equal(allIndices->size(), at.getNumElements()));
194 for (ArrayAttr subIdx : allIndices.value()) {
195 genWrite(loc, newArray, subIdx, *newResults, rewriter);
196 newResults++;
197 }
198 } else {
199 newResults++;
200 }
201 }
202 // erase the original CallOp
203 rewriter.eraseOp(oldCall);
204
205 return newCall;
206}

212void processBlockArgs(Block &entryBlock, ConversionPatternRewriter &rewriter) {
213 OpBuilder::InsertionGuard guard(rewriter);
214 rewriter.setInsertionPointToStart(&entryBlock);
215
216 for (unsigned i = 0; i < entryBlock.getNumArguments();) {
217 Value oldV = entryBlock.getArgument(i);
218 if (ArrayType at = splittableArray(oldV.getType())) {
219 Location loc = oldV.getLoc();
220 // Generate `CreateArrayOp` and replace uses of the argument with it.
221 auto newArray = rewriter.create<CreateArrayOp>(loc, at);
222 rewriter.replaceAllUsesWith(oldV, newArray);
223 // Remove the argument from the block
224 entryBlock.eraseArgument(i);
225 // For all indices in the ArrayType (i.e. the element count), generate a new block
226 // argument and a write of that argument to the new array.
227 std::optional<SmallVector<ArrayAttr>> allIndices = at.getSubelementIndices();
228 assert(allIndices); // follows from legal() check
229 assert(std::cmp_equal(allIndices->size(), at.getNumElements()));
230 for (ArrayAttr subIdx : allIndices.value()) {
231 BlockArgument newArg = entryBlock.insertArgument(i, at.getElementType(), loc);
232 genWrite(loc, newArray, subIdx, newArg, rewriter);
233 ++i;
234 }
235 } else {
236 ++i;
237 }
238 }
239}

247// If the operand has ArrayType, add N reads from the array to the `newOperands` list otherwise add
248// the original operand to the list.
249void processInputOperand(
250 Location loc, Value operand, SmallVector<Value> &newOperands,
251 ConversionPatternRewriter &rewriter
252) {
253 if (ArrayType at = splittableArray(operand.getType())) {
254 std::optional<SmallVector<ArrayAttr>> indices = at.getSubelementIndices();
255 assert(indices.has_value() && "passed earlier hasStaticShape() check");
256 for (ArrayAttr index : indices.value()) {
257 newOperands.push_back(genRead(loc, operand, index, rewriter));
258 }
259 } else {
260 newOperands.push_back(operand);
261 }
262}
263
264// For each operand with ArrayType, add N reads from the array and use those N values instead.
265void processInputOperands(
266 ValueRange operands, MutableOperandRange outputOpRef, Operation *op,
267 ConversionPatternRewriter &rewriter
268) {
269 SmallVector<Value> newOperands;
270 for (Value v : operands) {
271 processInputOperand(op->getLoc(), v, newOperands, rewriter);
272 }
273 rewriter.modifyOpInPlace(op, [&outputOpRef, &newOperands]() {
274 outputOpRef.assign(ValueRange(newOperands));
275 });
276}

280enum Direction {
282 SMALL_TO_LARGE,
284 LARGE_TO_SMALL,
285};
286
289template <Direction dir>
290inline void rewriteImpl(
291 ArrayAccessOpInterface op, ArrayType smallType, Value smallArr, Value largeArr,
292 ConversionPatternRewriter &rewriter
293) {
294 assert(smallType); // follows from legal() check
295 Location loc = op.getLoc();
296 MLIRContext *ctx = op.getContext();
297
298 ArrayAttr indexAsAttr = op.indexOperandsToAttributeArray();
299 assert(indexAsAttr); // follows from legal() check
300
301 // For all indices in the ArrayType (i.e. the element count), read from one array into the other
302 // (depending on direction flag).
303 std::optional<SmallVector<ArrayAttr>> subIndices = smallType.getSubelementIndices();
304 assert(subIndices); // follows from legal() check
305 assert(std::cmp_equal(subIndices->size(), smallType.getNumElements()));
306 for (ArrayAttr indexingTail : subIndices.value()) {
307 SmallVector<Attribute> joined;
308 joined.append(indexAsAttr.begin(), indexAsAttr.end());
309 joined.append(indexingTail.begin(), indexingTail.end());
310 ArrayAttr fullIndex = ArrayAttr::get(ctx, joined);
311
312 if constexpr (dir == Direction::SMALL_TO_LARGE) {
313 auto init = genRead(loc, smallArr, indexingTail, rewriter);
314 genWrite(loc, largeArr, fullIndex, init, rewriter);
315 } else if constexpr (dir == Direction::LARGE_TO_SMALL) {
316 auto init = genRead(loc, largeArr, fullIndex, rewriter);
317 genWrite(loc, smallArr, indexingTail, init, rewriter);
318 }
319 }
320}

367class SplitInitFromCreateArrayOp : public OpConversionPattern<CreateArrayOp> {
368public:
369 using OpConversionPattern<CreateArrayOp>::OpConversionPattern;
370
371 static bool legal(CreateArrayOp op) { return op.getElements().empty(); }
372
373 LogicalResult match(CreateArrayOp op) const override { return failure(legal(op)); }
374
375 void
376 rewrite(CreateArrayOp op, OpAdaptor adaptor, ConversionPatternRewriter &rewriter) const override {
377 // Remove elements from `op`
378 rewriter.modifyOpInPlace(op, [&op]() { op.getElementsMutable().clear(); });
379 // Generate an individual write for each initialization element
380 rewriter.setInsertionPointAfter(op);
381 Location loc = op.getLoc();
382 ArrayIndexGen idxGen = ArrayIndexGen::from(op.getType());
383 for (auto [i, init] : llvm::enumerate(adaptor.getElements())) {
384 // Convert the linear index 'i' into a multi-dim index
385 assert(std::cmp_less_equal(i, std::numeric_limits<int64_t>::max()));
386 std::optional<SmallVector<Value>> multiDimIdxVals =
387 idxGen.delinearize(static_cast<int64_t>(i), loc, rewriter);
388 // ASSERT: CreateArrayOp verifier ensures the number of elements provided matches the full
389 // linear array size so delinearization of `i` will not fail.
390 assert(multiDimIdxVals.has_value());
391 // Create the write
392 rewriter.create<WriteArrayOp>(loc, op.getResult(), ValueRange(*multiDimIdxVals), init);
393 }
394 }

397class SplitArrayInFuncDefOp : public OpConversionPattern<FuncDefOp> {
398public:
399 using OpConversionPattern<FuncDefOp>::OpConversionPattern;
400
401 inline static bool legal(FuncDefOp op) {
402 return !containsSplittableArrayType(op.getFunctionType());
403 }
404
405 LogicalResult match(FuncDefOp op) const override { return failure(legal(op)); }
406
407 void rewrite(FuncDefOp op, OpAdaptor, ConversionPatternRewriter &rewriter) const override {
408 // Update in/out types of the function to replace arrays with scalars
409 FunctionType oldTy = op.getFunctionType();
410 SmallVector<Type> newInputs = splitArrayType(oldTy.getInputs());
411 SmallVector<Type> newOutputs = splitArrayType(oldTy.getResults());
412 FunctionType newTy =
413 FunctionType::get(oldTy.getContext(), TypeRange(newInputs), TypeRange(newOutputs));
414 if (newTy == oldTy) {
415 return; // nothing to change
416 }
417 rewriter.modifyOpInPlace(op, [&op, &newTy]() { op.setFunctionType(newTy); });
418
419 // If the function has a body, ensure the entry block arguments match the function inputs.
420 if (Region *body = op.getCallableRegion()) {
421 Block &entryBlock = body->front();
422 if (std::cmp_equal(entryBlock.getNumArguments(), newInputs.size())) {
423 return; // nothing to change
424 }
425 processBlockArgs(entryBlock, rewriter);
426 }
427 }

467class ReplaceKnownArrayLengthOp : public OpConversionPattern<ArrayLengthOp> {
468public:
469 using OpConversionPattern<ArrayLengthOp>::OpConversionPattern;
470
472 static std::optional<llvm::APInt> getDimSizeIfKnown(Value dimIdx, ArrayType baseArrType) {
473 if (splittableArray(baseArrType)) {
474 llvm::APInt idxAP;
475 if (mlir::matchPattern(dimIdx, mlir::m_ConstantInt(&idxAP))) {
476 uint64_t idx64 = idxAP.getZExtValue();
477 assert(std::cmp_less_equal(idx64, std::numeric_limits<size_t>::max()));
478 Attribute dimSizeAttr = baseArrType.getDimensionSizes()[static_cast<size_t>(idx64)];
479 if (mlir::matchPattern(dimSizeAttr, mlir::m_ConstantInt(&idxAP))) {
480 return idxAP;
481 }
482 }
483 }
484 return std::nullopt;
485 }
486

488 // rewrite() can only work with constant dim size, i.e. must consider it legal otherwise
489 return !getDimSizeIfKnown(op.getDim(), op.getArrRefType()).has_value();
490 }
491
492 LogicalResult match(ArrayLengthOp op) const override { return failure(legal(op)); }
493
494 void
495 rewrite(ArrayLengthOp op, OpAdaptor adaptor, ConversionPatternRewriter &rewriter) const override {
496 ArrayType arrTy = dyn_cast<ArrayType>(adaptor.getArrRef().getType());
497 assert(arrTy); // must have array type per ODS spec of ArrayLengthOp
498 std::optional<llvm::APInt> len = getDimSizeIfKnown(adaptor.getDim(), arrTy);
499 assert(len.has_value()); // follows from legal() check
500 rewriter.replaceOpWithNewOp<arith::ConstantIndexOp>(op, llzk::fromAPInt(len.value()));
501 }

511class SplitArrayInFieldDefOp : public OpConversionPattern<FieldDefOp> {
512 SymbolTableCollection &tables;
513 FieldReplacementMap &repMapRef;
514
515public:
516 SplitArrayInFieldDefOp(
517 MLIRContext *ctx, SymbolTableCollection &symTables, FieldReplacementMap &fieldRepMap
518 )
519 : OpConversionPattern<FieldDefOp>(ctx), tables(symTables), repMapRef(fieldRepMap) {}
520
521 inline static bool legal(FieldDefOp op) { return !containsSplittableArrayType(op.getType()); }
522
523 LogicalResult match(FieldDefOp op) const override { return failure(legal(op)); }
524
525 void rewrite(FieldDefOp op, OpAdaptor, ConversionPatternRewriter &rewriter) const override {
526 StructDefOp inStruct = op->getParentOfType<StructDefOp>();
527 assert(inStruct);
528 LocalFieldReplacementMap &localRepMapRef = repMapRef[inStruct][op.getSymNameAttr()];
529
530 ArrayType arrTy = dyn_cast<ArrayType>(op.getType());
531 assert(arrTy); // follows from legal() check
532 auto subIdxs = arrTy.getSubelementIndices();
533 assert(subIdxs.has_value());
534 Type elemTy = arrTy.getElementType();
535
536 SymbolTable &structSymbolTable = tables.getSymbolTable(inStruct);
537 for (ArrayAttr idx : subIdxs.value()) {
538 // Create scalar version of the field
539 FieldDefOp newField =
540 rewriter.create<FieldDefOp>(op.getLoc(), op.getSymNameAttr(), elemTy, op.getColumn());
541 // Use SymbolTable to give it a unique name and store to the replacement map
542 localRepMapRef[idx] = std::make_pair(structSymbolTable.insert(newField), elemTy);
543 }
544 rewriter.eraseOp(op);
545 }

553template <typename ImplClass, typename FieldRefOpType, typename GenHeaderType>
554class SplitArrayInFieldRefOp : public OpConversionPattern<FieldRefOpType> {
555 SymbolTableCollection &tables;
556 const FieldReplacementMap &repMapRef;
557
558 // static check to ensure the functions are implemented in all subclasses
559 inline static void ensureImplementedAtCompile() {
560 static_assert(
561 sizeof(FieldRefOpType) == 0, "SplitArrayInFieldRefOp not implemented for requested type."
562 );
563 }
564
565protected:
566 using OpAdaptor = typename FieldRefOpType::Adaptor;
567
570 static GenHeaderType genHeader(FieldRefOpType, ConversionPatternRewriter &) {
571 ensureImplementedAtCompile();
572 assert(false && "unreachable");
573 }
574
577 static void
578 forIndex(Location, GenHeaderType, ArrayAttr, FieldInfo, OpAdaptor, ConversionPatternRewriter &) {
579 ensureImplementedAtCompile();
580 assert(false && "unreachable");
581 }
582
583public:
584 SplitArrayInFieldRefOp(
585 MLIRContext *ctx, SymbolTableCollection &symTables, const FieldReplacementMap &fieldRepMap
586 )
587 : OpConversionPattern<FieldRefOpType>(ctx), tables(symTables), repMapRef(fieldRepMap) {}
588
589 static bool legal(FieldRefOpType) {
590 ensureImplementedAtCompile();
591 assert(false && "unreachable");
592 }
593
594 LogicalResult match(FieldRefOpType op) const override { return failure(ImplClass::legal(op)); }
595
596 void rewrite(FieldRefOpType op, OpAdaptor adaptor, ConversionPatternRewriter &rewriter)
597 const override {
598 StructType tgtStructTy = llvm::cast<FieldRefOpInterface>(op.getOperation()).getStructType();
599 assert(tgtStructTy);
600 auto tgtStructDef = tgtStructTy.getDefinition(tables, op);
601 assert(succeeded(tgtStructDef));
602
603 GenHeaderType prefixResult = ImplClass::genHeader(op, rewriter);
604
605 const LocalFieldReplacementMap &idxToName =
606 repMapRef.at(tgtStructDef->get()).at(op.getFieldNameAttr().getAttr());
607 // Split the array field write into a series of read array + write scalar field
608 for (auto [idx, newField] : idxToName) {
609 ImplClass::forIndex(op.getLoc(), prefixResult, idx, newField, adaptor, rewriter);
610 }
611 rewriter.eraseOp(op);
612 }

663LogicalResult
664step1(ModuleOp modOp, SymbolTableCollection &symTables, FieldReplacementMap &fieldRepMap) {
665 MLIRContext *ctx = modOp.getContext();
666
667 RewritePatternSet patterns(ctx);
668
669 patterns.add<SplitArrayInFieldDefOp>(ctx, symTables, fieldRepMap);
670
671 ConversionTarget target(*ctx);
672 target.addLegalDialect<
673 LLZKDialect, array::ArrayDialect, boolean::BoolDialect, felt::FeltDialect,

729LogicalResult splitArrayCreateInit(ModuleOp modOp) {
730 SymbolTableCollection symTables;
731 FieldReplacementMap fieldRepMap;
732
733 // This is divided into 2 steps to simplify the implementation for field-related ops. The issue is
734 // that the conversions for field read/write expect the mapping of array index to field name+type
735 // to already be populated for the referenced field (although this could be computed on demand if
736 // desired but it complicates the implementation a bit).
737 if (failed(step1(modOp, symTables, fieldRepMap))) {
738 return failure();
739 }
740 return step2(modOp, symTables, fieldRepMap);
741}

743class ArrayToScalarPass : public llzk::array::impl::ArrayToScalarPassBase<ArrayToScalarPass> {
744 void runOnOperation() override {
745 ModuleOp module = getOperation();
746 // Separate array initialization from creation by removing the initalization list from
747 // CreateArrayOp and inserting the corresponding WriteArrayOp following it.
748 if (failed(splitArrayCreateInit(module))) {
749 signalPassFailure();
750 return;
751 }
752 OpPassManager nestedPM(ModuleOp::getOperationName());
753 // Use SROA (Destructurable* interfaces) to split each array with linear size N into N arrays of
754 // size 1. This is necessary because the mem2reg pass cannot deal with indexing and splitting up
755 // memory, i.e. it can only convert scalar memory access into SSA values.
756 nestedPM.addPass(createSROA());
757 // The mem2reg pass converts all of the size 1 array allocation and access into SSA values.
758 nestedPM.addPass(createMem2Reg());
759 if (failed(runPipeline(nestedPM, module))) {
760 signalPassFailure();
761 return;
762 }
763 }
764};
