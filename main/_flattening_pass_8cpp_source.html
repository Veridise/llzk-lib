88 void recordInstantiation(StructType oldType, StructType newType) {
89 // Assert invariant required by `structInstantiations`
90 assert(!isNullOrEmpty(oldType.getParams()));
91 assert(isNullOrEmpty(newType.getParams()));
92
93 auto forwardResult = structInstantiations.try_emplace(oldType, newType);
94 if (forwardResult.second) {
95 // Insertion was successful
96 // ASSERT: The reverse map does not contain this mapping either
97 assert(!reverseInstantiations.contains(newType));
98 reverseInstantiations[newType] = oldType;
99 // Set the modified flag
100 modified = true;
101 } else {
102 // ASSERT: If a mapping already existed for `oldType` it must be `newType`
103 assert(forwardResult.first->getSecond() == newType);
104 // ASSERT: The reverse mapping is already present as well
105 assert(reverseInstantiations.lookup(newType) == oldType);
106 }
107 assert(structInstantiations.size() == reverseInstantiations.size());
108 }
109
111 std::optional<StructType> getInstantiation(StructType oldType) const {
112 auto cachedResult = structInstantiations.find(oldType);
113 if (cachedResult != structInstantiations.end()) {
114 return cachedResult->second;
115 }
116 return std::nullopt;
117 }

120 DenseSet<SymbolRefAttr> getInstantiatedStructNames() const {
121 DenseSet<SymbolRefAttr> instantiatedNames;
122 for (const auto &[origRemoteTy, _] : structInstantiations) {
123 instantiatedNames.insert(origRemoteTy.getNameRef());
124 }
125 return instantiatedNames;

128 void reportDelayedDiagnostics(StructType newType, CallOp caller) {
129 auto res = delayedDiagnostics.find(newType);
130 if (res == delayedDiagnostics.end()) {
131 return;
132 }
133
134 DiagnosticEngine &engine = caller.getContext()->getDiagEngine();
135 for (Diagnostic &diag : res->second) {
136 // Update any notes referencing an UnknownLoc to use the CallOp location.
137 for (Diagnostic &note : diag.getNotes()) {
138 assert(note.getNotes().empty() && "notes cannot have notes attached");
139 if (llvm::isa<UnknownLoc>(note.getLocation())) {
140 note = std::move(Diagnostic(caller.getLoc(), note.getSeverity()).append(note.str()));
141 }
142 }
143 // Report. Based on InFlightDiagnostic::report().
144 engine.emit(std::move(diag));
145 }
146 // Emiting a Diagnostic consumes it (per DiagnosticEngine::emit) so remove them from the map.
147 // Unfortunately, this means if the key StructType is the result of instantiation at multiple
148 // `compute()` calls it will only be reported at one of those locations, not all.
149 delayedDiagnostics.erase(newType);
150 }

152 SmallVector<Diagnostic> &delayedDiagnosticSet(StructType newType) {
153 return delayedDiagnostics[newType];
154 }

158 bool isLegalConversion(Type oldType, Type newType, const char *patName) const {
159 std::function<bool(Type, Type)> checkInstantiations = [&](Type oTy, Type nTy) {
160 // Check if `oTy` is a struct with a known instantiation to `nTy`
161 if (StructType oldStructType = llvm::dyn_cast<StructType>(oTy)) {
162 // Note: The values in `structInstantiations` must be no-parameter struct types
163 // so there is no need for recursive check, simple equality is sufficient.
164 if (this->structInstantiations.lookup(oldStructType) == nTy) {
165 return true;
166 }
167 }
168 // Check if `nTy` is the result of a struct instantiation and if the pre-image of
169 // that instantiation (i.e. the parameterized version of the instantiated struct)
170 // is a more concrete unification of `oTy`.
171 if (StructType newStructType = llvm::dyn_cast<StructType>(nTy)) {
172 if (auto preImage = this->reverseInstantiations.lookup(newStructType)) {
173 if (isMoreConcreteUnification(oTy, preImage, checkInstantiations)) {
174 return true;
175 }
176 }
177 }
178 return false;
179 };
180
181 if (isMoreConcreteUnification(oldType, newType, checkInstantiations)) {
182 return true;
183 }
184 LLVM_DEBUG(llvm::dbgs() << "[" << patName << "] Cannot replace old type " << oldType
185 << " with new type " << newType
186 << " because it does not define a compatible and more concrete type.\n";
187 );
188 return false;
189 }
190
191 template <typename T, typename U>
192 inline bool areLegalConversions(T oldTypes, U newTypes, const char *patName) const {
193 return llvm::all_of(
194 llvm::zip_equal(oldTypes, newTypes),
195 [this, &patName](std::tuple<Type, Type> oldThenNew) {
196 return this->isLegalConversion(std::get<0>(oldThenNew), std::get<1>(oldThenNew), patName);
197 }
198 );
199 }

255// NOTE: This pattern will produce a compile error if `OpTy` does not define the general
256// `build(OpBuilder&, OperationState&, TypeRange, ValueRange, ArrayRef<NamedAttribute>)` function
257// because that function is required by the `replaceOpWithNewOp()` call.
258template <typename OpTy> class GeneralTypeReplacePattern : public OpConversionPattern<OpTy> {
259public:
260 using OpConversionPattern<OpTy>::OpConversionPattern;
261
262 LogicalResult matchAndRewrite(OpTy op, OpTy::Adaptor adaptor, ConversionPatternRewriter &rewriter)
263 const override {
264 const TypeConverter *converter = OpConversionPattern<OpTy>::getTypeConverter();
265 assert(converter);
266 // Convert result types
267 SmallVector<Type> newResultTypes;
268 if (failed(converter->convertTypes(op->getResultTypes(), newResultTypes))) {
269 return op->emitError("Could not convert Op result types.");
270 }
271 // ASSERT: 'adaptor.getAttributes()' is empty or subset of 'op->getAttrDictionary()' so the
272 // former can be ignored without losing anything.
273 assert(
274 adaptor.getAttributes().empty() ||
275 llvm::all_of(
276 adaptor.getAttributes(),
277 [d = op->getAttrDictionary()](NamedAttribute a) { return d.contains(a.getName()); }
278 )
279 );
280 // Convert any TypeAttr in the attribute list.
281 SmallVector<NamedAttribute> newAttrs(op->getAttrDictionary().getValue());
282 for (NamedAttribute &n : newAttrs) {
283 if (TypeAttr t = llvm::dyn_cast<TypeAttr>(n.getValue())) {
284 if (Type newType = converter->convertType(t.getValue())) {
285 n.setValue(TypeAttr::get(newType));
286 } else {
287 return op->emitError().append("Could not convert type in attribute: ", t);
288 }
289 }
290 }
291 // Build a new Op in place of the current one
292 replaceOpWithNewOp<OpTy>(
293 rewriter, op, TypeRange(newResultTypes), adaptor.getOperands(), ArrayRef(newAttrs)
294 );
295 return success();
296 }

367static bool defaultLegalityCheck(const TypeConverter &tyConv, Operation *op) {
368 // Check operand types and result types
369 if (!tyConv.isLegal(op)) {
370 return false;
371 }
372 // Check type attributes
373 // Extend lifetime of temporary to suppress warnings.
374 DictionaryAttr dictAttr = op->getAttrDictionary();
375 for (NamedAttribute n : dictAttr.getValue()) {
376 if (TypeAttr tyAttr = llvm::dyn_cast<TypeAttr>(n.getValue())) {
377 Type t = tyAttr.getValue();
378 if (FunctionType funcTy = llvm::dyn_cast<FunctionType>(t)) {
379 if (!tyConv.isSignatureLegal(funcTy)) {
380 return false;
381 }
382 } else {
383 if (!tyConv.isLegal(t)) {
384 return false;
385 }
386 }
387 }
388 }
389 return true;
390}

392// Default to true if the check is not for that particular operation type.
393template <typename Check> bool runCheck(Operation *op, Check check) {
394 if (auto specificOp =
395 mlir::dyn_cast_if_present<typename llvm::function_traits<Check>::template arg_t<0>>(op)) {
396 return check(specificOp);
397 }
398 return true;
399}

408template <typename... AdditionalOpTypes, typename... AdditionalChecks>
409ConversionTarget
410newConverterDefinedTarget(TypeConverter &tyConv, MLIRContext *ctx, AdditionalChecks &&...checks) {
411 ConversionTarget target = newBaseTarget(ctx);
412 auto inserter = [&](auto... opClasses) {
413 target.addDynamicallyLegalOp<decltype(opClasses)...>([&tyConv, &checks...](Operation *op) {
414 return defaultLegalityCheck(tyConv, op) && (runCheck<AdditionalChecks>(op, checks) && ...);
415 });
416 };
417 std::apply(inserter, OpClassesWithStructTypes.NoGeneralBuilder);
418 std::apply(inserter, OpClassesWithStructTypes.WithGeneralBuilder);
419 applyToMoreTypes<decltype(inserter), AdditionalOpTypes...>(inserter);
420 return target;
421}

444 class MappedTypeConverter : public TypeConverter {
445 StructType origTy;
446 StructType newTy;
447 const DenseMap<Attribute, Attribute> &paramNameToValue;
448
449 public:
450 MappedTypeConverter(
451 StructType originalType, StructType newType,
453 const DenseMap<Attribute, Attribute> &paramNameToInstantiatedValue
454 )
455 : TypeConverter(), origTy(originalType), newTy(newType),
456 paramNameToValue(paramNameToInstantiatedValue) {
457
458 addConversion([](Type inputTy) { return inputTy; });
459
460 addConversion([this](StructType inputTy) {
461 // Check for replacement of the full type
462 if (inputTy == this->origTy) {
463 return this->newTy;
464 }
465 // Check for replacement of parameter symbol names with concrete values
466 if (ArrayAttr inputTyParams = inputTy.getParams()) {
467 SmallVector<Attribute> updated;
468 for (Attribute a : inputTyParams) {
469 auto res = this->paramNameToValue.find(a);
470 updated.push_back((res != this->paramNameToValue.end()) ? res->second : a);
471 }
472 return StructType::get(
473 inputTy.getNameRef(), ArrayAttr::get(inputTy.getContext(), updated)
474 );
475 }
476 // Otherwise, return the type unchanged
477 return inputTy;
478 });
479
480 addConversion([this](ArrayType inputTy) {
481 // Check for replacement of parameter symbol names with concrete values
482 ArrayRef<Attribute> dimSizes = inputTy.getDimensionSizes();
483 if (!dimSizes.empty()) {
484 SmallVector<Attribute> updated;
485 for (Attribute a : dimSizes) {
486 auto res = this->paramNameToValue.find(a);
487 updated.push_back((res != this->paramNameToValue.end()) ? res->second : a);
488 }
489 return ArrayType::get(this->convertType(inputTy.getElementType()), updated);
490 }
491 // Otherwise, return the type unchanged
492 return inputTy;
493 });
494
495 addConversion([this](TypeVarType inputTy) -> Type {
496 // Check for replacement of parameter symbol name with a concrete type
497 auto res = this->paramNameToValue.find(inputTy.getNameRef());
498 if (res != this->paramNameToValue.end()) {
499 if (TypeAttr tyAttr = llvm::dyn_cast<TypeAttr>(res->second)) {
500 return tyAttr.getValue();
501 }
502 }
503 return inputTy;
504 });
505 }

533
534 SymbolUserHelper(
535 TypeConverter &converter, MLIRContext *ctx, unsigned Benefit,
536 const DenseMap<Attribute, Attribute> &paramNameToInstantiatedValue
537 )
538 : OpConversionPattern<Op>(converter, ctx, Benefit),

540
541 public:
542 using OpAdaptor = typename mlir::OpConversionPattern<Op>::OpAdaptor;

545
546 virtual LogicalResult handleDefaultRewrite(
547 Attribute, Op op, OpAdaptor, ConversionPatternRewriter &, Attribute a

550 }
551
552 LogicalResult
553 matchAndRewrite(Op op, OpAdaptor adaptor, ConversionPatternRewriter &rewriter) const override {
554 auto res = this->paramNameToValue.find(getNameAttr(op));
555 if (res == this->paramNameToValue.end()) {
556 return op->emitOpError("missing instantiation");
557 }
558 llvm::TypeSwitch<Attribute, LogicalResult> TS(res->second);
559 llvm::TypeSwitch<Attribute, LogicalResult> *ptr = &TS;
560
561 ((ptr = &(ptr->template Case<HandledAttrs>([&](HandledAttrs a) {
562 return static_cast<const Impl *>(this)->handleRewrite(res->first, op, adaptor, rewriter, a);
563 }))),
564 ...);
565
566 return TS.Default([&](Attribute a) {
567 return handleDefaultRewrite(res->first, op, adaptor, rewriter, a);
568 });
569 }

593 LogicalResult handleRewrite(
594 Attribute sym, ConstReadOp op, OpAdaptor, ConversionPatternRewriter &rewriter, IntegerAttr a
595 ) const {
596 APInt attrValue = a.getValue();
597 Type origResTy = op.getType();
598 if (llvm::isa<FeltType>(origResTy)) {
599 replaceOpWithNewOp<FeltConstantOp>(
600 rewriter, op, FeltConstAttr::get(getContext(), attrValue)
601 );
602 return success();
603 }
604
605 if (llvm::isa<IndexType>(origResTy)) {
606 replaceOpWithNewOp<arith::ConstantIndexOp>(rewriter, op, fromAPInt(attrValue));
607 return success();
608 }
609
610 if (origResTy.isSignlessInteger(1)) {
611 // Treat 0 as false and any other value as true (but give a warning if it's not 1)
612 if (attrValue.isZero()) {
613 replaceOpWithNewOp<arith::ConstantIntOp>(rewriter, op, false, origResTy);
614 return success();
615 }
616 if (!attrValue.isOne()) {
617 Location opLoc = op.getLoc();
618 Diagnostic diag(opLoc, DiagnosticSeverity::Warning);
619 diag << "Interpretting non-zero value " << stringWithoutType(a) << " as true";
620 if (getContext()->shouldPrintOpOnDiagnostic()) {
621 diag.attachNote(opLoc) << "see current operation: " << *op;
622 }
623 diag.attachNote(UnknownLoc::get(getContext()))
624 << "when instantiating '" << StructDefOp::getOperationName() << "' parameter \""
625 << sym << "\" for this call";
626 diagnostics.push_back(std::move(diag));
627 }
628 replaceOpWithNewOp<arith::ConstantIntOp>(rewriter, op, true, origResTy);
629 return success();
630 }
631 return LogicalResult(op->emitOpError().append("unexpected result type ", origResTy));
632 }

682 static inline DenseMap<Attribute, Attribute>
683 buildNameToValueMap(ArrayAttr paramNames, ArrayRef<Attribute> paramInstantiations) {
684 // pre-conditions
685 assert(!isNullOrEmpty(paramNames));
686 assert(paramNames.size() == paramInstantiations.size());
687 // Map parameter names to instantiated values
688 DenseMap<Attribute, Attribute> ret;
689 for (size_t i = 0, e = paramNames.size(); i < e; ++i) {
690 ret[paramNames[i]] = paramInstantiations[i];
691 }
692 return ret;

695 FailureOr<StructType> genClone(StructType typeAtCaller, ArrayRef<Attribute> typeAtCallerParams) {
696 // Find the StructDefOp for the original StructType
697 FailureOr<SymbolLookupResult<StructDefOp>> r = typeAtCaller.getDefinition(symTables, rootMod);
698 if (failed(r)) {
699 return failure(); // getDefinition() already emits a sufficient error message
700 }
701 StructDefOp origStruct = r->get();
702
703 // Clone the original struct, apply the new name, and remove the parameters.
704 StructDefOp newStruct = origStruct.clone();
705 newStruct.setSymName(
706 typeAtCaller.getNameRef().getLeafReference().str() + "_" + shortString(typeAtCallerParams)
707 );
708 newStruct.setConstParamsAttr(ArrayAttr {});
709
710 // Insert 'newStruct' into the parent ModuleOp of the original StructDefOp. Use the
711 // `SymbolTable::insert()` function directly so that the name will be made unique.
712 ModuleOp parentModule = llvm::cast<ModuleOp>(origStruct.getParentOp());
713 symTables.getSymbolTable(parentModule).insert(newStruct, Block::iterator(origStruct));
714 // Retrieve the new type AFTER inserting since the name may be appended to make it unique.
715 StructType newRemoteType = newStruct.getType();
716
717 // Within the new struct, replace all references to the original struct's type (i.e. the
718 // locally-parameterized version) with the new flattened (i.e. no parameters) struct's type,
719 // and replace all uses of the struct parameters with the concrete values.
720 MLIRContext *ctx = rootMod.getContext();
721 StructType typeAtDef = origStruct.getType();
722 DenseMap<Attribute, Attribute> nameToValueMap =
723 buildNameToValueMap(typeAtDef.getParams(), typeAtCallerParams);
724 MappedTypeConverter tyConv(typeAtDef, newRemoteType, nameToValueMap);
725 ConversionTarget target =
726 newConverterDefinedTarget<EmitEqualityOp>(tyConv, ctx, tableOffsetIsntSymbol);
727
728 target.addIllegalOp<ConstReadOp>();
729
730 RewritePatternSet patterns = newGeneralRewritePatternSet<EmitEqualityOp>(tyConv, ctx, target);
731 patterns.add<ClonedStructCallOpPattern>(tyConv, ctx);
732 patterns.add<ClonedStructConstReadOpPattern>(
733 tyConv, ctx, nameToValueMap, tracker_.delayedDiagnosticSet(newRemoteType)
734 );
735 patterns.add<ClonedStructFieldReadOpPattern>(tyConv, ctx, nameToValueMap);
736 if (failed(applyFullConversion(newStruct, target, std::move(patterns)))) {
737 return failure();
738 }
739 return newRemoteType;
740 }
741
742public:
743 StructCloner(ConversionTracker &tracker, ModuleOp root)
744 : tracker_(tracker), rootMod(root), symTables() {}
745
746 FailureOr<StructType> createInstantiatedClone(StructType orig) {
747 if (ArrayAttr params = orig.getParams()) {
748 // If all parameters are concrete values (Integer or Type), then replace with a
749 // no-parameter StructType referencing the de-parameterized struct.
750 if (llvm::all_of(params, isConcreteAttr<>)) {
751 FailureOr<StructType> res = genClone(orig, params.getValue());
752 if (succeeded(res)) {
753 return res.value();
754 }
755 }
756 }
757 return failure();
758 }

916struct AffineMapFolder {
917 struct Input {
918 OperandRangeRange mapOpGroups;
919 DenseI32ArrayAttr dimsPerGroup;
920 ArrayRef<Attribute> paramsOfStructTy;
921 };
922
923 struct Output {
924 SmallVector<SmallVector<Value>> mapOpGroups;
925 SmallVector<int32_t> dimsPerGroup;
926 SmallVector<Attribute> paramsOfStructTy;
927 };
928
929 static inline SmallVector<ValueRange> getConvertedMapOpGroups(Output out) {
930 return llvm::map_to_vector(out.mapOpGroups, [](const SmallVector<Value> &grp) {
931 return ValueRange(grp);
932 });
933 }
934
935 static LogicalResult
936 fold(PatternRewriter &rewriter, const Input &in, Output &out, Operation *op, const char *aspect) {
937 if (in.mapOpGroups.empty()) {
938 // No affine map operands so nothing to do
939 return failure();
940 }
941
942 assert(in.mapOpGroups.size() <= in.paramsOfStructTy.size());
943 assert(std::cmp_equal(in.mapOpGroups.size(), in.dimsPerGroup.size()));
944
945 size_t idx = 0; // index in `mapOpGroups`, i.e. the number of AffineMapAttr encountered
946 for (Attribute sizeAttr : in.paramsOfStructTy) {
947 if (AffineMapAttr m = dyn_cast<AffineMapAttr>(sizeAttr)) {
948 ValueRange currMapOps = in.mapOpGroups[idx++];
949 LLVM_DEBUG(
950 llvm::dbgs() << "[AffineMapFolder] currMapOps: " << debug::toStringList(currMapOps)
951 << '\n'
952 );
953 SmallVector<OpFoldResult> currMapOpsCast = getAsOpFoldResult(currMapOps);
954 LLVM_DEBUG(
955 llvm::dbgs() << "[AffineMapFolder] currMapOps as fold results: "
956 << debug::toStringList(currMapOpsCast) << '\n'
957 );
958 if (auto constOps = Step3_InstantiateAffineMaps::getConstantIntValues(currMapOpsCast)) {
959 SmallVector<Attribute> result;
960 bool hasPoison = false; // indicates divide by 0 or mod by <1
961 auto constAttrs = llvm::map_to_vector(*constOps, [&rewriter](int64_t v) -> Attribute {
962 return rewriter.getIndexAttr(v);
963 });
964 LogicalResult foldResult = m.getAffineMap().constantFold(constAttrs, result, &hasPoison);
965 if (hasPoison) {
966 LLVM_DEBUG(op->emitRemark().append(
967 "Cannot fold affine_map for ", aspect, " ", out.paramsOfStructTy.size(),
968 " due to divide by 0 or modulus with negative divisor"
969 ));
970 return failure();
971 }
972 if (failed(foldResult)) {
973 LLVM_DEBUG(op->emitRemark().append(
974 "Folding affine_map for ", aspect, " ", out.paramsOfStructTy.size(), " failed"
975 ));
976 return failure();
977 }
978 if (result.size() != 1) {
979 LLVM_DEBUG(op->emitRemark().append(
980 "Folding affine_map for ", aspect, " ", out.paramsOfStructTy.size(), " produced ",
981 result.size(), " results but expected 1"
982 ));
983 return failure();
984 }
985 assert(!llvm::isa<AffineMapAttr>(result[0]) && "not converted");
986 out.paramsOfStructTy.push_back(result[0]);
987 continue;
988 }
989 // If affine but not foldable, preserve the map ops
990 out.mapOpGroups.emplace_back(currMapOps);
991 out.dimsPerGroup.push_back(in.dimsPerGroup[idx - 1]); // idx was already incremented
992 }
993 // If not affine and foldable, preserve the original
994 out.paramsOfStructTy.push_back(sizeAttr);
995 }
996 assert(idx == in.mapOpGroups.size() && "all affine_map not processed");
997 assert(
998 in.paramsOfStructTy.size() == out.paramsOfStructTy.size() &&
999 "produced wrong number of dimensions"
1000 );
1001
1002 return success();
1003 }
1004};

1296 llvm::dbgs() << "[InstantiateAtCallOpCompute] oldRetTy: " << oldRetTy << '\n';
1297 });
1298 ArrayAttr params = oldRetTy.getParams();
1299 if (isNullOrEmpty(params)) {
1300 // nothing to do if the StructType is not parameterized
1301 return failure();
1302 }
1303
1304 AffineMapFolder::Output out;
1305 AffineMapFolder::Input in = {
1306 op.getMapOperands(),
1307 op.getNumDimsPerMapAttr(),
1308 params.getValue(),
1309 };
1310 if (!in.mapOpGroups.empty()) {
1311 // If there are affine map operands, attempt to fold them to a constant.
1312 if (failed(AffineMapFolder::fold(rewriter, in, out, op, "struct parameter"))) {
1313 return failure();
1314 }
1315 LLVM_DEBUG({
1316 llvm::dbgs() << "[InstantiateAtCallOpCompute] folded affine_map in result type params\n";
1317 });
1318 } else {
1319 // If there are no affine map operands, attempt to refine the result type of the CallOp using
1320 // the function argument types and the type of the target function.
1321 auto callArgTypes = op.getArgOperands().getTypes();
1322 if (callArgTypes.empty()) {
1323 // no refinement posible if no function arguments
1324 return failure();
1325 }
1326 SymbolTableCollection tables;
1327 auto lookupRes = lookupTopLevelSymbol<FuncDefOp>(tables, op.getCalleeAttr(), op);
1328 if (failed(lookupRes)) {
1329 return failure();
1330 }
1331 if (failed(instantiateViaTargetType(in, out, callArgTypes, lookupRes->get()))) {
1332 return failure();
1333 }
1334 LLVM_DEBUG({
1335 llvm::dbgs() << "[InstantiateAtCallOpCompute] propagated instantiations via symrefs in "
1336 "result type params: "
1337 << debug::toStringList(out.paramsOfStructTy) << '\n';
1338 });
1339 }
1340
1341 StructType newRetTy = StructType::get(oldRetTy.getNameRef(), out.paramsOfStructTy);
1342 if (newRetTy == oldRetTy) {
1343 // nothing changed
1344 return failure();
1345 }
1346 // The `newRetTy` is computed via instantiateViaTargetType() which can only preserve the
1347 // original Attribute or convert to a concrete attribute via the unification process. Thus, if
1348 // the conversion here is illegal it means there is a type conflict within the LLZK code that
1349 // prevents instantiation of the struct with the requested type.
1350 if (!tracker_.isLegalConversion(oldRetTy, newRetTy, "InstantiateAtCallOpCompute")) {
1351 return rewriter.notifyMatchFailure(op, [&](Diagnostic &diag) {
1352 diag.append(
1353 "result type mismatch: due to struct instantiation, expected type ", newRetTy,
1354 ", but found ", oldRetTy
1355 );
1356 });
1357 }
1358 LLVM_DEBUG(
1359 llvm::dbgs() << "[InstantiateAtCallOpCompute] instantiating " << oldRetTy << " as "
1360 << newRetTy << " in \"" << op << "\"\n"
1361 );
1362 replaceOpWithNewOp<CallOp>(
1363 rewriter, op, TypeRange {newRetTy}, op.getCallee(),
1364 AffineMapFolder::getConvertedMapOpGroups(out), out.dimsPerGroup, op.getArgOperands()
1365 );
1366 return success();
1367 }

1372 inline LogicalResult instantiateViaTargetType(
1373 const AffineMapFolder::Input &in, AffineMapFolder::Output &out,
1374 OperandRange::type_range callArgTypes, FuncDefOp targetFunc
1375 ) const {
1376 assert(targetFunc.isStructCompute()); // since `op.calleeIsStructCompute()`
1377 ArrayAttr targetResTyParams = targetFunc.getSingleResultTypeOfCompute().getParams();
1378 assert(!isNullOrEmpty(targetResTyParams)); // same cardinality as `in.paramsOfStructTy`
1379 assert(in.paramsOfStructTy.size() == targetResTyParams.size()); // verifier ensures this
1380
1381 if (llvm::all_of(in.paramsOfStructTy, [](Attribute a) { return isConcreteAttr<>(a); })) {
1382 // Nothing can change if everything is already concrete
1383 return failure();
1384 }
1385
1386 LLVM_DEBUG({
1387 llvm::dbgs() << '[' << __FUNCTION__ << ']'
1388 << " call arg types: " << debug::toStringList(callArgTypes) << '\n';
1389 llvm::dbgs() << '[' << __FUNCTION__ << ']' << " target func arg types: "
1390 << debug::toStringList(targetFunc.getArgumentTypes()) << '\n';
1391 llvm::dbgs() << '[' << __FUNCTION__ << ']'
1392 << " struct params @ call: " << debug::toStringList(in.paramsOfStructTy) << '\n';
1393 llvm::dbgs() << '[' << __FUNCTION__ << ']'
1394 << " target struct params: " << debug::toStringList(targetResTyParams) << '\n';
1395 });
1396
1397 UnificationMap unifications;
1398 bool unifies = typeListsUnify(targetFunc.getArgumentTypes(), callArgTypes, {}, &unifications);
1399 assert(unifies && "should have been checked by verifiers");
1400
1401 LLVM_DEBUG({
1402 llvm::dbgs() << '[' << __FUNCTION__ << ']'
1403 << " unifications of arg types: " << debug::toStringList(unifications) << '\n';
1404 });
1405
1406 // Check for LHS SymRef (i.e. from the target function) that have RHS concrete Attributes (i.e.
1407 // from the call argument types) without any struct parameters (because the type with concrete
1408 // struct parameters will be used to instantiate the target struct rather than the fully
1409 // flattened struct type resulting in type mismatch of the callee to target) and perform those
1410 // replacements in the `targetFunc` return type to produce the new result type for the CallOp.
1411 SmallVector<Attribute> newReturnStructParams = llvm::map_to_vector(
1412 llvm::zip_equal(targetResTyParams.getValue(), in.paramsOfStructTy),
1413 [&unifications](std::tuple<Attribute, Attribute> p) {
1414 Attribute fromCall = std::get<1>(p);
1415 // Preserve attributes that are already concrete at the call site. Otherwise attempt to lookup
1416 // non-parameterized concrete unification for the target struct parameter symbol.
1417 if (!isConcreteAttr<>(fromCall)) {
1418 Attribute fromTgt = std::get<0>(p);
1419 LLVM_DEBUG({
1420 llvm::dbgs() << "[instantiateViaTargetType] fromCall = " << fromCall << '\n';
1421 llvm::dbgs() << "[instantiateViaTargetType] fromTgt = " << fromTgt << '\n';
1422 });
1423 assert(llvm::isa<SymbolRefAttr>(fromTgt));
1424 auto it = unifications.find(std::make_pair(llvm::cast<SymbolRefAttr>(fromTgt), Side::LHS));
1425 if (it != unifications.end()) {
1426 Attribute unifiedAttr = it->second;
1427 LLVM_DEBUG({
1428 llvm::dbgs() << "[instantiateViaTargetType] unifiedAttr = " << unifiedAttr << '\n';
1429 });
1430 if (unifiedAttr && isConcreteAttr<false>(unifiedAttr)) {
1431 return unifiedAttr;
1432 }
1433 }
1434 }
1435 return fromCall;
1436 }
1437 );
1438
1439 out.paramsOfStructTy = newReturnStructParams;
1440 assert(out.paramsOfStructTy.size() == in.paramsOfStructTy.size() && "post-condition");
1441 assert(out.mapOpGroups.empty() && "post-condition");
1442 assert(out.dimsPerGroup.empty() && "post-condition");
1443 return success();
1444 }
1445};

1469template <typename OpTy> class EraseOpPattern : public OpConversionPattern<OpTy> {
1470public:
1471 EraseOpPattern(MLIRContext *ctx) : OpConversionPattern<OpTy>(ctx) {}
1472
1473 LogicalResult

1480LogicalResult run(ModuleOp modOp, const ConversionTracker &tracker) {
1481 FailureOr<ModuleOp> topRoot = getTopRootModule(modOp);
1482 if (failed(topRoot)) {
1483 return failure();
1484 }
1485
1486 // Use a conversion to erase instantiated structs if they have no other references.
1487 //
1488 // TODO: There's a chance the "no other references" criteria will leave some behind when running
1489 // only a single pass of this because they may reference each other. Maybe I can check if the
1490 // references are only located within another struct in the list, but would have to do a deep
1491 // deep lookup to ensure no references and avoid infinite loop back on self. The CallGraphAnalysis
1492 // is not sufficient because it looks only at calls but there could be (although unlikely) a
1493 // FieldDefOp referencing a struct type despite having no calls to that struct's functions.
1494 //
1495 // TODO: There's another scenario that leaves some behind. Once a StructDefOp is visited and
1496 // considered legal, that decision cannot be reversed. Hence, StructDefOp that become illegal only
1497 // after removing another one that uses it will not be removed. See
1498 // test/Dialect/LLZK/instantiate_structs_affine_pass.llzk
1499 // One idea is to use one of the `SymbolTable::getSymbolUses` functions starting from a struct
1500 // listed in `instantiatedNames` to determine if it is reachable from some other struct that is
1501 // NOT listed there and remove it if not. For efficiency, this reachability information can be
1502 // pre-computed and or cached.
1503 //
1504 DenseSet<SymbolRefAttr> instantiatedNames = tracker.getInstantiatedStructNames();
1505 auto isLegalStruct = [&](bool emitWarning, StructDefOp op) {
1506 if (instantiatedNames.contains(op.getType().getNameRef())) {
1507 if (!hasUsesWithin(op, *topRoot)) {
1508 // Parameterized struct with no uses is illegal, i.e. should be removed.
1509 return false;
1510 }
1511 if (emitWarning) {
1512 op.emitWarning("Parameterized struct still has uses!").report();
1513 }
1514 }
1515 return true;
1516 };
1517
1518 // Perform the conversion, i.e. remove StructDefOp that were instantiated and are unused.
1519 MLIRContext *ctx = modOp.getContext();
1520 RewritePatternSet patterns(ctx);
1521 patterns.add<EraseOpPattern<StructDefOp>>(ctx);
1522 ConversionTarget target = newBaseTarget(ctx);
1523 // TODO: Here are some thoughts from LLZK planning:
1524 // - Flattening could remove all structs that are not instantiated and the backend would have to
1525 // not run flattening if it wants to keep any templated structs.
1526 // - Alternatively, flattening could define different conversion targets at runtime based on
1527 // config flags to indicate if some should be flattened and others should not. This could be a
1528 // flag that indicates allow/restrict globally or based on some struct criteria, like names.
1529 //
1530 // target.addIllegalDialect<polymorphic::PolymorphicDialect>();
1531 target.addDynamicallyLegalOp<StructDefOp>(std::bind_front(isLegalStruct, false));
1532 if (failed(applyFullConversion(modOp, target, std::move(patterns)))) {
1533 return failure();
1534 }
1535
1536 // Warn about any structs that were instantiated but still have uses elsewhere.
1537 modOp->walk([&](StructDefOp op) {
1538 isLegalStruct(true, op);
1539 return WalkResult::skip(); // StructDefOp cannot be nested
1540 });
1541
1542 return success();
1543}

1560 signalPassFailure();

1577 signalPassFailure();

1589 llvm::dbgs() << "=====================================================================\n";

1594 } while (tracker.isModified());

1600 "instantiated versions\n";
