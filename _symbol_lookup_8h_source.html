141template <typename T>
142inline mlir::FailureOr<SymbolLookupResult<T>> lookupSymbolIn(
143 mlir::SymbolTableCollection &tables, mlir::SymbolRefAttr symbol, Within &&lookupWithin,
144 mlir::Operation *origin
145) {
146 auto found = lookupSymbolIn(tables, symbol, std::move(lookupWithin), origin);
147 if (mlir::failed(found)) {
148 return mlir::failure(); // lookupSymbolIn() already emits a sufficient error message
149 }
150 // Keep a copy of the op ptr in case we need it for displaying diagnostics
151 mlir::Operation *op = found->get();
152 // ... since the untyped result gets moved here into a typed result.
153 SymbolLookupResult<T> ret(std::move(*found));
154 if (!ret) {
155 return origin->emitError() << "symbol \"" << symbol << "\" references a '" << op->getName()
156 << "' but expected a '" << T::getOperationName() << "'";
157 }
158 return ret;
159}
