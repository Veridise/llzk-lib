43template <typename FieldUserOp> SymbolRefAttr getFullFieldSymbol(FieldUserOp op) {
44 FlatSymbolRefAttr fieldSym = op.getFieldNameAttr();
45 auto structType = dyn_cast<StructType>(op.getComponent().getType());
46 ensure(structType != nullptr, "given op type must operate on a struct type");
47 SymbolRefAttr structSym = structType.getNameRef(); // this is fully qualified
48 return appendLeaf(structSym, fieldSym);
49}

51class UnusedDeclarationEliminationPass
52 : public llzk::impl::UnusedDeclarationEliminationPassBase<UnusedDeclarationEliminationPass> {

56 struct PassContext {
57 DenseMap<SymbolRefAttr, StructDefOp> symbolToStruct;
58 DenseMap<StructDefOp, SymbolRefAttr> structToSymbol;
59
60 const SymbolRefAttr &getSymbol(StructDefOp s) const { return structToSymbol.at(s); }
61 StructDefOp getStruct(const SymbolRefAttr &sym) const { return symbolToStruct.at(sym); }
62
63 static PassContext populate(ModuleOp modOp) {
64 PassContext ctx;
65
66 modOp.walk<WalkOrder::PreOrder>([&ctx](StructDefOp structDef) {
67 auto structSymbolRes = getPathFromTopRoot(structDef);
68 ensure(succeeded(structSymbolRes), "failed to lookup struct symbol");
69 SymbolRefAttr structSym = *structSymbolRes;
70 ctx.symbolToStruct[structSym] = structDef;
71 ctx.structToSymbol[structDef] = structSym;
72 });
73 return ctx;
74 }
75 };

77 void runOnOperation() override {
78 PassContext ctx = PassContext::populate(getOperation());
79 // First, remove unused fields. This may allow more structs to be removed,
80 // if their final remaining uses are as types for unused fields.
81 removeUnusedFields(ctx);
82
83 // Last, remove unused structs if configured
84 if (removeStructs) {
85 removeUnusedStructs(ctx);
86 }
87 }

92 void removeUnusedFields(PassContext &ctx) {
93 ModuleOp modOp = getOperation();
94
95 // Map fully-qualified field symbols -> field ops
96 DenseMap<SymbolRefAttr, FieldDefOp> fields;
97 for (auto &[structDef, structSym] : ctx.structToSymbol) {
98 structDef.walk([&](FieldDefOp field) {
99 // We don't consider public fields in the Main component for removal,
100 // as these are output values and removing them would result in modifying
101 // the overall circuit interface.
102 if (!structDef.isMainComponent() || !field.hasPublicAttr()) {
103 SymbolRefAttr fieldSym =
104 appendLeaf(structSym, FlatSymbolRefAttr::get(field.getSymNameAttr()));
105 fields[fieldSym] = field;
106 }
107 });
108 }
109
110 // Remove all fields that are read.
111 modOp.walk([&](FieldReadOp readf) {
112 SymbolRefAttr readFieldSym = getFullFieldSymbol(readf);
113 fields.erase(readFieldSym);
114 });
115
116 // Remove all writes that reference the remaining fields, as these writes
117 // are now known to only update write-only fields.
118 modOp.walk([&](FieldWriteOp writef) {
119 SymbolRefAttr writtenField = getFullFieldSymbol(writef);
120 if (fields.contains(writtenField)) {
121 // We need not check the users of a writef, since it produces no results.
122 LLVM_DEBUG(
123 llvm::dbgs() << "Removing write " << writef << " to write-only field " << writtenField
124 << '\n'
125 );
126 writef.erase();
127 }
128 });
129
130 // Finally, erase the remaining fields.
131 for (auto &[_, fieldDef] : fields) {
132 LLVM_DEBUG(llvm::dbgs() << "Removing field " << fieldDef << '\n');
133 fieldDef->erase();
134 }
135 }

141 void removeUnusedStructs(PassContext &ctx) {
142 DenseMap<StructDefOp, DenseSet<StructDefOp>> uses;
143 DenseMap<StructDefOp, DenseSet<StructDefOp>> usedBy;
144
145 // initialize both maps with empty sets so we can identify unused structs
146 for (auto &[structDef, _] : ctx.structToSymbol) {
147 uses[structDef] = {};
148 usedBy[structDef] = {};
149 }
150
151 getOperation().walk([&](Operation *op) {
152 auto structParent = op->getParentOfType<StructDefOp>();
153 if (structParent == nullptr) {
154 return WalkResult::advance();
155 }
156
157 auto tryAddUse = [&](Type ty) {
158 if (auto structTy = dyn_cast<StructType>(ty)) {
159 // This name ref is required to be fully qualified
160 SymbolRefAttr sym = structTy.getNameRef();
161 StructDefOp refStruct = ctx.getStruct(sym);
162 if (refStruct != structParent) {
163 uses[structParent].insert(refStruct);
164 usedBy[refStruct].insert(structParent);
165 }
166 }
167 };
168
169 // LLZK requires fully-qualified references to struct symbols. So, we
170 // simply need to look for the struct symbol within this op's symbol uses.
171
172 // Check operands
173 for (Value operand : op->getOperands()) {
174 tryAddUse(operand.getType());
175 }
176
177 // Check results
178 for (Value result : op->getResults()) {
179 tryAddUse(result.getType());
180 }
181
182 // Check block arguments
183 for (Region &region : op->getRegions()) {
184 for (Block &block : region) {
185 for (BlockArgument arg : block.getArguments()) {
186 tryAddUse(arg.getType());
187 }
188 }
189 }
190
191 // Check attributes
192 for (const auto &namedAttr : op->getAttrs()) {
193 namedAttr.getValue().walk([&](TypeAttr typeAttr) { tryAddUse(typeAttr.getValue()); });
194 }
195
196 return WalkResult::advance();
197 });
198
199 SmallVector<StructDefOp> unusedStructs;
200
201 auto updateUnusedStructs = [&]() {
202 for (auto &[structDef, users] : usedBy) {
203 if (users.empty() && !structDef.isMainComponent()) {
204 unusedStructs.push_back(structDef);
205 }
206 }
207 };
208
209 updateUnusedStructs();
210
211 while (!unusedStructs.empty()) {
212 StructDefOp unusedStruct = unusedStructs.back();
213 unusedStructs.pop_back();
214
215 // See what structs are being used by this unused struct
216 for (auto usedStruct : uses[unusedStruct]) {
217 // The usedStruct is no longer used by the unusedStruct
218 usedBy[usedStruct].erase(unusedStruct);
219 }
220
221 // Remove the unused struct from both maps and the IR
222 usedBy.erase(unusedStruct);
223 uses.erase(unusedStruct);
224 unusedStruct->erase();
225
226 // Check to see if we've created any more unused structs after we process
227 // all existing known unused structs (to avoid double processing).
228 if (unusedStructs.empty()) {
229 updateUnusedStructs();
230 }
231 }
232 }
233};
