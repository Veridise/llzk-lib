114enum class Side { EMPTY = 0, LHS, RHS, TOMB };
115static inline mlir::raw_ostream &operator<<(mlir::raw_ostream &os, const Side &val) {
116 switch (val) {
117 case Side::EMPTY:
118 os << "EMPTY";
119 break;
120 case Side::TOMB:
121 os << "TOMB";
122 break;
123 case Side::LHS:
124 os << "LHS";
125 break;
126 case Side::RHS:

198template <typename Iter1, typename Iter2>
199inline bool typeListsUnify(
200 Iter1 lhs, Iter2 rhs, mlir::ArrayRef<llvm::StringRef> rhsReversePrefix = {},
201 UnificationMap *unifications = nullptr
202) {
203 return (lhs.size() == rhs.size()) &&
204 std::equal(lhs.begin(), lhs.end(), rhs.begin(), [&](mlir::Type a, mlir::Type b) {
205 return typesUnify(a, b, rhsReversePrefix, unifications);
206 });
207}
208
209template <typename Iter1, typename Iter2>
210inline bool singletonTypeListsUnify(
211 Iter1 lhs, Iter2 rhs, mlir::ArrayRef<llvm::StringRef> rhsReversePrefix = {},
212 UnificationMap *unifications = nullptr
213) {
214 return lhs.size() == 1 && rhs.size() == 1 &&
215 typesUnify(lhs.front(), rhs.front(), rhsReversePrefix, unifications);
216}
