28SymbolLookupResultUntyped
29lookupSymbolRec(SymbolTableCollection &tables, SymbolRefAttr symbol, Operation *symTableOp) {
30 // First try a direct lookup via the SymbolTableCollection. Must use a low-level lookup function
31 // in order to properly account for modules that were added due to inlining IncludeOp.
32 {
33 SmallVector<Operation *, 4> symbolsFound;
34 if (succeeded(tables.lookupSymbolIn(symTableOp, symbol, symbolsFound))) {
35 SymbolLookupResultUntyped ret(symbolsFound.back());
36 for (auto it = symbolsFound.rbegin(); it != symbolsFound.rend(); ++it) {
37 Operation *op = *it;
38 if (op->hasAttr(LANG_ATTR_NAME)) {
39 ret.trackIncludeAsName(op->getAttrOfType<StringAttr>(SymbolTable::getSymbolAttrName()));
40 }
41 }
42 return ret;
43 }
44 }
45 // Otherwise, check if the reference can be found by manually doing a lookup for each part of
46 // the reference in turn, traversing through IncludeOp symbols by parsing the included file.
47 if (Operation *rootOp = tables.lookupSymbolIn(symTableOp, symbol.getRootReference())) {
48 if (IncludeOp rootOpInc = llvm::dyn_cast<IncludeOp>(rootOp)) {
49 FailureOr<OwningOpRef<ModuleOp>> otherMod = rootOpInc.openModule();
50 if (succeeded(otherMod)) {
51 // Create a temporary SymbolTableCollection for caching the external symbols from the
52 // included module rather than adding these symbols to the existing SymbolTableCollection
53 // because it has no means of removing entries from its internal map and it is not safe to
54 // leave the dangling pointers in that map after the external module has been freed.
55 SymbolTableCollection external;
56 auto result = lookupSymbolRec(external, getTailAsSymbolRefAttr(symbol), otherMod->get());
57 if (result) {
58 result.manage(std::move(*otherMod), std::move(external));
59 result.trackIncludeAsName(rootOpInc.getSymName());
60 }
61 return result;
62 }
63 } else if (ModuleOp rootOpMod = llvm::dyn_cast<ModuleOp>(rootOp)) {
64 return lookupSymbolRec(tables, getTailAsSymbolRefAttr(symbol), rootOpMod);
65 }
66 }
67 // Otherwise, return empty result
68 return SymbolLookupResultUntyped();
69}

87void SymbolLookupResultUntyped::manage(
88 OwningOpRef<ModuleOp> &&ptr, SymbolTableCollection &&tables
89) {
90 // This may be called multiple times for the same result Operation but we only need to store the
91 // resources from the first call because that call will contain the final ModuleOp loaded in a
92 // chain of IncludeOp and that is the one which contains the result Operation*.
93 if (!managedResources) {
94 managedResources = std::make_shared<std::pair<OwningOpRef<ModuleOp>, SymbolTableCollection>>(
95 std::make_pair(std::move(ptr), std::move(tables))
96 );
97 }

116FailureOr<SymbolLookupResultUntyped> Within::lookup(
117 SymbolTableCollection &tables, SymbolRefAttr symbol, Operation *origin, bool reportMissing
118) && {
119 if (SymbolLookupResultUntyped *priorRes = std::get_if<SymbolLookupResultUntyped>(&this->from)) {
120 //---- Lookup within an existing result ----//
121 // Use the symbol table from prior result if available, otherwise use the parameter.
122 SymbolTableCollection *cachedTablesForRes = priorRes->getSymbolTableCache();
123 if (!cachedTablesForRes) {
124 cachedTablesForRes = &tables;
125 }
126 if (auto found = lookupSymbolRec(*cachedTablesForRes, symbol, priorRes->op)) {
127 assert(!found.managedResources && "should not have loaded additional modules");
128 // TODO: not quite sure following is true. If not, the result should contain
129 // `priorRes.includeSymNameStack` followed by `found.includeSymNameStack`.
130 assert(found.includeSymNameStack.empty() && "should not have loaded additional modules");
131 // Move stuff from 'priorRes' to the new result
132 found.managedResources = std::move(priorRes->managedResources);
133 found.includeSymNameStack = std::move(priorRes->includeSymNameStack);
134 return found;
135 }
136 } else {
137 //---- Lookup from a given operation or root (if nullptr) ----//
138 Operation *lookupFrom = std::get<Operation *>(this->from);
139 if (!lookupFrom) {
140 FailureOr<ModuleOp> root = getRootModule(origin);
141 if (failed(root)) {
142 return failure(); // getRootModule() already emits a sufficient error message
143 }
144 lookupFrom = root.value();
145 }
146 if (auto found = lookupSymbolRec(tables, symbol, lookupFrom)) {
147 return found;
148 }
149 }
150 // Handle the case where it was not found
151 if (reportMissing) {
152 return origin->emitOpError() << "references unknown symbol \"" << symbol << "\"";
153 } else {
154 return failure();
155 }
156}
