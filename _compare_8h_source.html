29template <OpComparable Op> mlir::FailureOr<bool> isLocationLess(const Op &l, const Op &r) {
30 Op &lhs = const_cast<Op &>(l);
31 Op &rhs = const_cast<Op &>(r);
32 // Try sorting by location first, then name.
33 auto lhsLoc = llvm::dyn_cast<mlir::FileLineColLoc>(lhs.getOperation()->getLoc());
34 auto rhsLoc = llvm::dyn_cast<mlir::FileLineColLoc>(rhs.getOperation()->getLoc());
35 if (lhsLoc && rhsLoc) {
36 auto filenameCmp = lhsLoc.getFilename().compare(rhsLoc.getFilename());
37 return filenameCmp < 0 || (filenameCmp == 0 && lhsLoc.getLine() < rhsLoc.getLine()) ||
38 (filenameCmp == 0 && lhsLoc.getLine() == rhsLoc.getLine() &&
39 lhsLoc.getColumn() < rhsLoc.getColumn());
40 }
41 return mlir::failure();
42}
43
44template <OpComparable Op> struct OpLocationLess {
45 bool operator()(const Op &l, const Op &r) const { return isLocationLess(l, r).value_or(false); }
46};
47
48template <NamedOpComparable Op> struct NamedOpLocationLess {
49 bool operator()(const Op &l, const Op &r) const {
50 auto res = isLocationLess(l, r);
51 if (mlir::succeeded(res)) {
52 return res.value();
53 }
54
55 Op &lhs = const_cast<Op &>(l);
56 Op &rhs = const_cast<Op &>(r);
57 return lhs.getName().compare(rhs.getName()) < 0;
58 }
59};
